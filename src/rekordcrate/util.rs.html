15/// Enumerates errors returned by this library.
16#[derive(Error, Debug)]
17#[non_exhaustive]
18pub enum RekordcrateError {
19    /// Represents a failure to decode a DeviceSQL string.
20    #[error(transparent)]
21    StringError(#[from] StringError),
22
23    /// Represents a failure to parse input.
24    #[error(transparent)]
25    ParseError(#[from] binrw::Error),
26
27    /// Represents an `std::io::Error`.
28    #[error(transparent)]
29    IOError(#[from] std::io::Error),
30}
31
32/// Type alias for results where the error is a `RekordcrateError`.
33pub type RekordcrateResult<T> = std::result::Result<T, RekordcrateError>;

94/// align given value to the alignment requirements by the given type.
95#[must_use]
96pub const fn align_by(alignment: u64, mut offset: u64) -> u64 {
97    // This is technically dependent on the compile time ABI
98    // but for x86 (which this is likely compiled on), we should be able
99    // to assume that
100    // the alignment of a type is just the size of its largest
101    // member. That likely matches the assumptions made for the 32-bit
102    // MCU (Renesas R8A77240D500BG) built into the different CDJ-2000 variants.
103    // In either way, its better to overshoot the alignment
104    // than to undershoot it. For CDJ-3000s, this assumption
105    // is likely also correct since they use a 64-bit ARM CPU (Renesas R8A774C0HA01BG)
106    if !offset.is_multiple_of(alignment) {
107        offset += alignment - (offset % alignment);
108    }
109    offset
110}

112#[binrw(little)]
113#[derive(Clone, Copy, Debug, PartialEq, Eq, Default)]
114#[br(import(limit: usize))]
115/// Represents explicit padding bytes in a binary structure.
116pub struct ExplicitPadding(
117    // use offset + a couple bytes as a limit (though this is rather a heuristic)
118    #[br(parse_with = Self::guess_padding, args(limit))]
119    #[bw(ignore)]
120    #[bw(pad_after=self.0)]
121    pub usize,
122);
123
124use binrw::{io::Read, BinResult};
125impl ExplicitPadding {
126    #[binrw::parser(reader)]
127    fn guess_padding(limit: usize) -> BinResult<usize> {
128        let before = reader.stream_position()?;
129        let mut count = 0;
130        // We never expect to read many bytes here, so this is fine
131        #[allow(clippy::unbuffered_bytes)]
132        for byte in reader.bytes() {
133            if byte? != 0 {
134                break;
135            }
136            // to avoid scanning an entire page after the last row,
137            // we limit here
138            // if the limit is reached, assume end of page and no padding
139            // used
140            if limit != 0 && count == limit {
141                count = 0;
142                break;
143            }
144            count += 1;
145        }
146        if reader.stream_position()? != before {
147            // don't consume the non-zero byte we just read if we read anything at all
148            reader.seek_relative(-1)?;
149        }
150        Ok(count)
151    }
152}
153
154impl binrw::meta::ReadEndian for ExplicitPadding {
155    const ENDIAN: binrw::meta::EndianKind = binrw::meta::EndianKind::None;
156}
157impl binrw::meta::WriteEndian for ExplicitPadding {
158    const ENDIAN: binrw::meta::EndianKind = binrw::meta::EndianKind::None;
159}
160
161impl From<usize> for ExplicitPadding {
162    fn from(value: usize) -> Self {
163        Self(value)
164    }
165}

168pub(crate) mod testing {
169    use binrw::{
170        meta::{ReadEndian, WriteEndian},

172        Endian,
173    };
174    use pretty_assertions::assert_eq;
175    use pretty_hex::pretty_hex;

177    macro_rules! assert_eq_hex {
178        ($cond:expr, $expected:expr) => {
179            assert_eq!(pretty_hex($cond), pretty_hex($expected));
180        };
181    }
182
183    pub fn test_roundtrip_with_args<'a, T>(
184        bin: &[u8],
185        obj: T,
186        read_args: <T as binrw::BinRead>::Args<'a>,
187        write_args: <T as binrw::BinWrite>::Args<'a>,
188    ) where
189        <T as binrw::BinRead>::Args<'a>: Clone,
190        <T as binrw::BinWrite>::Args<'a>: Clone,
191        T: BinRead + BinWrite + PartialEq + core::fmt::Debug + ReadEndian + WriteEndian,
192    {
193        let endian = Endian::NATIVE;
194        // T->binary
195        let mut writer = binrw::io::Cursor::new(Vec::with_capacity(bin.len()));
196        obj.write_options(&mut writer, endian, write_args.clone())
197            .unwrap();
198        assert_eq_hex!(&writer.get_ref(), &bin);
199        // T->binary->T
200        writer.set_position(0);
201        let parsed = T::read_options(&mut writer, endian, read_args.clone()).unwrap();
202        assert_eq!(parsed, obj);
203        // binary->T
204        let mut cursor = binrw::io::Cursor::new(bin);
205        let parsed = T::read_options(&mut cursor, endian, read_args.clone()).unwrap();
206        assert_eq!(parsed, obj);
207        // binary->T->binary
208        writer.set_position(0);
209        parsed
210            .write_options(&mut writer, endian, write_args.clone())
211            .unwrap();
212        assert_eq_hex!(&bin, &writer.get_ref());
213    }
214
215    pub fn test_roundtrip<'a, T>(bin: &[u8], obj: T)
216    where
217        <T as binrw::BinRead>::Args<'a>: Default + Clone,
218        <T as binrw::BinWrite>::Args<'a>: Default + Clone,
219        T: BinRead + BinWrite + PartialEq + core::fmt::Debug + ReadEndian + WriteEndian,
220    {
221        test_roundtrip_with_args(
222            bin,
223            obj,
224            <T as binrw::BinRead>::Args::default(),
225            <T as binrw::BinWrite>::Args::default(),
226        );
227    }
228}
