31    IntegrityError(&'static str),
32
33    /// Represents an `std::io::Error`.
34    #[error(transparent)]
35    IOError(#[from] std::io::Error),
36
37    /// Represents an `std::io::Error`.
38    #[error("component not loaded")]
39    NotLoadedError,
40}

114/// Parses a sequence of values with `BinRead` from the offsets provided by the iterator.
115pub(crate) fn parse_at_offsets<Offset, Value, T, Args, It, Reader>(
116    it: It,
117) -> impl FnOnce(&mut Reader, Endian, Args) -> BinResult<T>
118where
119    Offset: IntoSeekFrom,
120    Value: for<'a> BinRead<Args<'a> = Args>,
121    T: FromIterator<(Offset, Value)>,
122    Args: Clone,
123    It: IntoIterator<Item = Offset>,
124    Reader: Read + Seek,
125{
126    move |reader, endian, args| {
127        let base = reader.stream_position()?;
128        it.into_iter()
129            .map(|offset| {
130                reader.seek(offset.into_seek_from())?;
131                let v = Value::read_options(reader, endian, args.clone())?;
132                // Restore position after each item.
133                reader.seek(SeekFrom::Start(base))?;
134                Ok((offset, v))
135            })
136            .collect()
137    }
138}

140/// Writes a sequence of values with `BinWrite` at the offsets provided by the iterator.
141pub(crate) fn write_at_offsets<Offset, Value, T, Args, Writer>(
142    t: &T,
143    writer: &mut Writer,
144    endian: Endian,
145    args: Args,
146) -> BinResult<()>
147where
148    Offset: IntoSeekFrom,
149    Value: for<'a> BinWrite<Args<'a> = Args>,
150    for<'a> &'a T: IntoIterator<Item = (&'a Offset, &'a Value)>,
151    Args: Clone,
152    Writer: Write + Seek,
153{
154    let base = writer.stream_position()?;
155    t.into_iter().try_for_each(|(offset, v)| {
156        writer.seek(offset.into_seek_from())?;
157        v.write_options(writer, endian, args.clone())?;
158        // Restore position after each item.
159        writer.seek(SeekFrom::Start(base))?;
160        Ok(())
161    })
162}

164/// align given value to the alignment requirements by the given type.
165#[must_use]
166pub const fn align_by(alignment: u64, mut offset: u64) -> u64 {
167    // This is technically dependent on the compile time ABI
168    // but for x86 (which this is likely compiled on), we should be able
169    // to assume that
170    // the alignment of a type is just the size of its largest
171    // member. That likely matches the assumptions made for the 32-bit
172    // MCU (Renesas R8A77240D500BG) built into the different CDJ-2000 variants.
173    // In either way, its better to overshoot the alignment
174    // than to undershoot it. For CDJ-3000s, this assumption
175    // is likely also correct since they use a 64-bit ARM CPU (Renesas R8A774C0HA01BG)
176    if !offset.is_multiple_of(alignment) {
177        offset += alignment - (offset % alignment);
178    }
179    offset
180}

182pub(crate) trait ArrayPolyfills<T, const N: usize> {
183    fn try_map_polyfill<U, E>(self, f: impl FnMut(T) -> Result<U, E>) -> Result<[U; N], E>;
184    fn zip_polyfill<U>(self, other: [U; N]) -> [(T, U); N];
185}
186
187impl<T, const N: usize> ArrayPolyfills<T, N> for [T; N] {
188    fn try_map_polyfill<U, E>(self, f: impl FnMut(T) -> Result<U, E>) -> Result<[U; N], E> {
189        let results = self.map(f);
190        let has_error = results.iter().any(|r| r.is_err());
191        if has_error {
192            return Err(results.into_iter().find_map(|r| r.err()).unwrap());
193        }
194        Ok(results.map(|r| match r {
195            Ok(v) => v,
196            Err(_) => unreachable!(),
197        }))
198    }
199
200    fn zip_polyfill<U>(self, other: [U; N]) -> [(T, U); N] {
201        // With a bit of luck the compiler will optimize all of this to avoid
202        // all of the Option packing and unpacking.
203        let mut a = self.map(Some);
204        let mut b = other.map(Some);
205        let mut result: [Option<(T, U)>; N] = [const { None }; _];
206        for i in 0..N {
207            result[i] = Some((a[i].take().unwrap(), b[i].take().unwrap()));
208        }
209        result.map(|opt| match opt {
210            Some(v) => v,
211            None => unreachable!(),
212        })
213    }
214}

227        ($cond:expr, $expected:expr) => {
228            assert_eq!(pretty_hex($cond), pretty_hex($expected));

232    pub fn test_write<'a, T>(bin: &[u8], obj: &T, args: <T as binrw::BinWrite>::Args<'a>)
233    where
234        T: BinWrite + PartialEq + core::fmt::Debug + WriteEndian,
235    {
236        let endian = Endian::NATIVE;
237        // T->binary
238        let mut writer = binrw::io::Cursor::new(Vec::with_capacity(bin.len()));
239        obj.write_options(&mut writer, endian, args).unwrap();
240        assert_eq_hex!(&writer.get_ref(), &bin);
241    }
242
243    pub fn test_roundtrip_with_args<'a, T>(
244        bin: &[u8],
245        obj: T,
246        read_args: <T as binrw::BinRead>::Args<'a>,
247        write_args: <T as binrw::BinWrite>::Args<'a>,
248    ) where
249        <T as binrw::BinRead>::Args<'a>: Clone,
250        <T as binrw::BinWrite>::Args<'a>: Clone,
251        T: BinRead + BinWrite + PartialEq + core::fmt::Debug + ReadEndian + WriteEndian,
252    {
253        let endian = Endian::NATIVE;
254        // T->binary
255        let mut writer = binrw::io::Cursor::new(Vec::with_capacity(bin.len()));
256        obj.write_options(&mut writer, endian, write_args.clone())
257            .unwrap();
258        assert_eq_hex!(&writer.get_ref(), &bin);
259        // T->binary->T
260        writer.set_position(0);
261        let parsed = T::read_options(&mut writer, endian, read_args.clone()).unwrap();
262        assert_eq!(parsed, obj);
263        // binary->T
264        let mut cursor = binrw::io::Cursor::new(bin);
265        let parsed = T::read_options(&mut cursor, endian, read_args.clone()).unwrap();
266        assert_eq!(parsed, obj);
267        // binary->T->binary
268        writer.set_position(0);
269        parsed
270            .write_options(&mut writer, endian, write_args.clone())
271            .unwrap();
272        assert_eq_hex!(&bin, &writer.get_ref());
273    }

290#[cfg(test)]
291mod test {
292    use super::*;
293    use std::{collections::BTreeMap, io::Cursor};
294
295    #[test]
296    fn test_parse_at_offsets() {
297        // Little-endian u16 values at offsets 0, 2, 5, 7 with garbage 4th byte.
298        const DATA: &[u8] = &[0x01, 0x00, 0x02, 0x00, 0xFF, 0x03, 0x00, 0x04, 0x00];
299        const OFFSETS: &[u64] = &[0, 7, 2, 5];
300        let mut cursor = Cursor::new(DATA);
301        let serialized: BTreeMap<u64, u16> =
302            parse_at_offsets(OFFSETS.iter().copied())(&mut cursor, Endian::Little, ()).unwrap();
303        let expected = BTreeMap::from([(0u64, 1u16), (7, 4), (2, 2), (5, 3)]);
304        assert_eq!(serialized, expected);
305
306        // We also expect the cursor position to be restored.
307        assert_eq!(cursor.stream_position().unwrap(), 0);
308    }
309
310    #[test]
311    fn test_write_at_offsets() {
312        // Little-endian u16 values at offsets 0, 2, 5, 7 with skipped 4th byte.
313        const EXPECTED: &[u8] = &[0x01, 0x00, 0x02, 0x00, 0x00, 0x03, 0x00, 0x04, 0x00];
314        let serialized: BTreeMap<u64, u16> =
315            vec![(0, 1), (7, 4), (2, 2), (5, 3)].into_iter().collect();
316        let mut cursor = Cursor::new(Vec::with_capacity(EXPECTED.len()));
317        write_at_offsets(&serialized, &mut cursor, Endian::Little, ()).unwrap();
318        assert_eq!(&cursor.get_ref()[..], EXPECTED);
319
320        // We also expect the cursor position to be restored.
321        assert_eq!(cursor.stream_position().unwrap(), 0);
322    }
