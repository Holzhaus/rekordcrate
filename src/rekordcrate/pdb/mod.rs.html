125/// Points to a table page and can be used to calculate the page's file offset by multiplying it
126/// with the page size (found in the file header).
127#[binrw]
128#[derive(Clone, Debug, PartialEq, Eq, PartialOrd)]
129#[brw(little)]
130pub struct PageIndex(u32);
131
132impl PageIndex {
133    /// Calculate the absolute file offset of the page in the PDB file for the given `page_size`.
134    #[must_use]
135    pub fn offset(&self, page_size: u32) -> u64 {
136        u64::from(self.0) * u64::from(page_size)
137    }
138}

140/// Tables are linked lists of pages containing rows of a single type, which are organized
141/// into groups.
142#[binrw]
143#[derive(Debug, PartialEq, Eq, Clone)]
144#[brw(little)]
145#[brw(import(db_type: DatabaseType))]
146pub struct Table {
147    /// Identifies the type of rows that this table contains.
148    #[brw(args(db_type))]
149    pub page_type: PageType,
150    /// Unknown field, maybe links to a chain of empty pages if the database is ever garbage
151    /// collected (?).
152    #[allow(dead_code)]
153    empty_candidate: u32,
154    /// Index of the first page that belongs to this table.
155    ///
156    /// *Note:* The first page apparently does not contain any rows. If the table is non-empty, the
157    /// actual row data can be found in the pages after.
158    pub first_page: PageIndex,
159    /// Index of the last page that belongs to this table.
160    pub last_page: PageIndex,
161}

163/// The PDB header structure, including the list of tables.
164#[binrw]
165#[derive(Debug, PartialEq, Eq, Clone)]
166#[brw(little)]
167#[brw(import(db_type: DatabaseType))]
168pub struct Header {
169    /// Unknown purpose, perhaps an unoriginal signature, seems to always have the value 0.
170    #[br(temp, assert(unknown1 == 0))]
171    #[bw(calc = 0u32)]
172    unknown1: u32,
173    /// Size of a single page in bytes.
174    ///
175    /// The byte offset of a page can be calculated by multiplying a page index with this value.
176    pub page_size: u32,
177    /// Number of tables.
178    #[br(temp)]
179    #[bw(calc = tables.len().try_into().expect("too many tables"))]
180    num_tables: u32,
181    /// Unknown field, not used as any `empty_candidate`, points past end of file.
182    #[allow(dead_code)]
183    next_unused_page: PageIndex,
184    /// Unknown field.
185    #[allow(dead_code)]
186    unknown: u32,
187    /// Always incremented by at least one, sometimes by two or three.
188    pub sequence: u32,
189    /// The gap seems to be always zero.
190    #[br(temp, assert(gap == 0))]
191    #[bw(calc = 0u32)]
192    gap: u32,
193    /// Each table is a linked list of pages containing rows of a particular type.
194    #[br(count = num_tables, args {inner: (db_type,)})]
195    #[bw(args(db_type))]
196    pub tables: Vec<Table>,
197}
198
199impl Header {
200    /// Returns pages for the given Table.
201    pub fn read_pages<R: Read + Seek>(
202        &self,
203        reader: &mut R,
204        _: Endian,
205        args: (&PageIndex, &PageIndex, DatabaseType),
206    ) -> BinResult<Vec<Page>> {
207        let endian = Endian::Little;
208        let (first_page, last_page, db_type) = args;
209
210        let mut pages = vec![];
211        let mut page_index = first_page.clone();
212        loop {
213            let page_offset = SeekFrom::Start(page_index.offset(self.page_size));
214            reader.seek(page_offset).map_err(binrw::Error::Io)?;
215            let page = Page::read_options(reader, endian, (self.page_size, db_type))?;
216            let is_last_page = &page.page_index == last_page;
217            page_index = page.next_page.clone();
218            pages.push(page);
219
220            if is_last_page {
221                break;
222            }
223        }
224        Ok(pages)
225    }
226}

239fn write_page_contents<W: Write + Seek>(
240    row_groups: &Vec<RowGroup>,
241    writer: &mut W,
242    endian: Endian,
243    args: (u32,),
244) -> BinResult<()> {
245    let (page_size,) = args;
246
247    let header_end_pos = writer.stream_position()?;
248
249    let mut relative_row_offset: u64 = 0;
250
251    // Seek to the very end of the page
252    writer.seek(SeekFrom::Current((page_size - Page::HEADER_SIZE).into()))?;
253
254    for row_group in row_groups {
255        relative_row_offset = row_group.write_options_and_get_row_offset(
256            writer,
257            endian,
258            (header_end_pos, relative_row_offset),
259        )?;
260    }
261    Ok(())
262}

264/// A table page.
265///
266/// Each page consists of a header that contains information about the type, number of rows, etc.,
267/// followed by the data section that holds the row data. Each row needs to be located using an
268/// offset found in the page footer at the end of the page.
269#[binrw]
270#[derive(Debug, PartialEq)]
271#[brw(little, import(page_size: u32, db_type: DatabaseType))]
272pub struct Page {
273    /// Stream position at the beginning of the page; used to compute heap base for standalone buffers.
274    #[br(temp, parse_with = current_offset)]
275    #[bw(ignore)]
276    page_start_pos: u64,
277    /// Magic signature for pages (must be 0).
278    #[br(temp, assert(magic == 0u32))]
279    #[bw(calc = 0u32)]
280    magic: u32,
281    /// Index of the page.
282    ///
283    /// Should match the index used for lookup and can be used to verify that the correct page was loaded.
284    pub page_index: PageIndex,
285    /// Type of information that the rows of this page contain.
286    ///
287    /// Should match the page type of the table that this page belongs to.
288    #[brw(args(db_type))]
289    pub page_type: PageType,
290    /// Index of the next page with the same page type.
291    ///
292    /// If this page is the last one of that type, the page index stored in the field will point
293    /// past the end of the file.
294    pub next_page: PageIndex,
295    /// Unknown field.
296    #[allow(dead_code)]
297    unknown1: u32,
298    /// Unknown field.
299    #[allow(dead_code)]
300    unknown2: u32,
301    /// Number of rows in this table (8-bit version).
302    ///
303    /// Used if `num_rows_large` not greater than this value and not equal to `0x1FFF`, which means
304    /// that the number of rows fits into a single byte.
305    pub num_rows_small: u8,
306    /// Unknown field.
307    ///
308    /// According to [@flesniak](https://github.com/flesniak):
309    /// > a bitmask (first track: 32)
310    #[allow(dead_code)]
311    unknown3: u8,
312    /// Unknown field.
313    ///
314    /// According to [@flesniak](https://github.com/flesniak):
315    /// > often 0, sometimes larger, esp. for pages with high real_entry_count (e.g. 12 for 101 entries)
316    #[allow(dead_code)]
317    unknown4: u8,
318    /// Page flags.
319    ///
320    /// According to [@flesniak](https://github.com/flesniak):
321    /// > strange pages: 0x44, 0x64; otherwise seen: 0x24, 0x34
322    page_flags: PageFlags,
323    /// Free space in bytes in the data section of the page (excluding the row offsets in the page footer).
324    pub free_size: u16,
325    /// Used space in bytes in the data section of the page.
326    pub used_size: u16,
327    /// Unknown field.
328    ///
329    /// According to [@flesniak](https://github.com/flesniak):
330    /// > (0->1: 2)
331    #[allow(dead_code)]
332    unknown5: u16,
333    /// Number of rows in this table (16-bit version).
334    ///
335    /// Used when the number of rows does not fit into a single byte. In that case,`num_rows_large`
336    /// is greater than `num_rows_small`, but is not equal to `0x1FFF`.
337    pub num_rows_large: u16,
338    /// Unknown field.
339    #[allow(dead_code)]
340    unknown6: u16,
341    /// Unknown field.
342    ///
343    /// According to [@flesniak](https://github.com/flesniak):
344    /// > always 0, except 1 for history pages, num entries for strange pages?"
345    #[allow(dead_code)]
346    unknown7: u16,
347    /// Number of rows in this page.
348    ///
349    /// **Note:** This is a virtual field and not actually read from the file.
350    #[br(temp, calc = Self::calculate_num_rows(num_rows_small, num_rows_large))]
351    #[bw(ignore)]
352    num_rows: u16,
353    /// Number of rows groups in this page.
354    ///
355    /// **Note:** This is a virtual field and not actually read from the file.
356    #[br(temp, calc = num_rows.div_ceil(RowGroup::MAX_ROW_COUNT as u16))]
357    #[bw(ignore)]
358    num_row_groups: u16,
359    /// The offset at which the row data for this page are located.
360    ///
361    /// **Note:** This is a virtual field and not actually read from the file.
362    #[br(temp, calc = page_start_pos + u64::from(Self::HEADER_SIZE))]
363    #[bw(ignore)]
364    page_heap_offset: u64,
365    /// Row groups belonging to this page.
366    #[br(seek_before(SeekFrom::Current(Self::heap_padding_size(page_size, num_row_groups).into())))]
367    #[br(args {count: num_row_groups.into(), inner: (page_type, page_heap_offset)})]
368    #[br(map(|mut vec: Vec<RowGroup>| {vec.reverse(); vec}))]
369    #[br(if(num_row_groups > 0 && page_flags.page_has_data()))]
370    #[bw(write_with = write_page_contents, args(page_size))]
371    pub row_groups: Vec<RowGroup>,
372}
373
374impl Page {
375    /// Size of the page header in bytes.
376    pub const HEADER_SIZE: u32 = 0x28;
377
378    /// Calculate the size of the empty space between the header and the footer.
379    fn heap_padding_size(page_size: u32, num_row_groups: u16) -> u32 {
380        // Size of all row offsets
381        let row_groups_footer_size = u32::from(num_row_groups) * RowGroup::BINARY_SIZE;
382        page_size - Self::HEADER_SIZE - row_groups_footer_size
383    }
384
385    #[must_use]
386    /// Returns `true` if the page actually contains row data.
387    pub fn has_data(&self) -> bool {
388        self.page_flags.page_has_data()
389    }
390
391    #[must_use]
392    /// Number of rows on this page.
393    ///
394    /// Note that this number includes rows that have been flagged as missing by the row group.
395    pub fn num_rows(&self) -> u16 {
396        Self::calculate_num_rows(self.num_rows_small, self.num_rows_large)
397    }
398
399    fn calculate_num_rows(num_rows_small: u8, num_rows_large: u16) -> u16 {
400        if num_rows_large > num_rows_small.into() && num_rows_large != 0x1fff {
401            num_rows_large
402        } else {
403            num_rows_small.into()
404        }
405    }
406}

408/// A group of row indices, which are built backwards from the end of the page. Holds up to sixteen
409/// row offsets, along with a bit mask that indicates whether each row is actually present in the
410/// table.
411#[binread]
412#[br(import(page_type: PageType, page_heap_position: u64))]
413#[derive(Debug, Clone)]
414pub struct RowGroup {
415    /// An offset which points to a row in the table, whose actual presence is controlled by one of the
416    /// bits in `row_present_flags`. This instance allows the row itself to be lazily loaded, unless it
417    /// is not present, in which case there is no content to be loaded.
418    // rustc doesn't seem to recognize that this is used below, ignore for now
419    #[allow(dead_code)]
420    row_offsets: [u16; Self::MAX_ROW_COUNT],
421    row_presence_flags: u16,
422    /// Unknown field, probably padding.
423    ///
424    /// Apparently this is not always zero, so it might also be something different.
425    unknown: u16,
426    // build rows from offsets collected above
427    #[br(seek_before=SeekFrom::Start(page_heap_position))]
428    #[br(args(page_type))]
429    #[br(parse_with = binrw::file_ptr::parse_from_iter(Self::present_rows_offsets(&row_offsets, row_presence_flags)))]
430    #[br(restore_position)] // ensure the parser points to just after this instance, this is important
431    /// Access rows in this RowGroup
432    rows: Vec<Row>,
433}
434
435impl RowGroup {
436    const MAX_ROW_COUNT: usize = 16;
437    const BINARY_SIZE: u32 = (Self::MAX_ROW_COUNT as u32) * 2 + 4; // size the serialized structure
438
439    /// Get all rows present in this rowgroup
440    #[must_use]
441    pub fn present_rows(&self) -> &[Row] {
442        &self.rows
443    }
444    // TODO(Swiftb0y): Add a new error category for user APIs and add the correct
445    // error herer
446    #[allow(clippy::result_unit_err)]
447    /// Add a row to this rowgroup
448    pub fn add_row(&mut self, row: Row) -> Result<(), ()> {
449        if self.rows.len() >= Self::MAX_ROW_COUNT {
450            return Err(());
451        }
452        self.row_presence_flags |= 1 << self.rows.len() as u16;
453        self.rows.push(row);
454        Ok(())
455    }
456
457    fn present_rows_offsets(
458        row_offsets: &[u16; Self::MAX_ROW_COUNT],
459        row_presence_flags: u16,
460    ) -> impl Iterator<Item = u16> + '_ {
461        row_offsets
462            .iter()
463            .rev()
464            .enumerate()
465            .filter_map(move |(i, row_offset)| {
466                (row_presence_flags & (1 << i) != 0).then_some(*row_offset)
467            })
468    }
469}
470
471impl PartialEq for RowGroup {
472    fn eq(&self, other: &Self) -> bool {
473        self.rows == other.rows
474    }
475}

477impl RowGroup {
478    // This helper function now lives in the main impl block for RowGroup
479    // Assumes we point just past the rowgroup we're trying to write.
480    fn write_options_and_get_row_offset<W: Write + Seek>(
481        &self,
482        writer: &mut W,
483        endian: Endian,
484        args: (u64, u64),
485    ) -> binrw::BinResult<u64> {
486        let (heap_start, relative_row_offset) = args;
487
488        let rows_to_write_count = self.present_rows().len();
489
490        // The number of flags set should match the number of present rows.
491        if rows_to_write_count != self.row_presence_flags.count_ones() as usize {
492            return Err(binrw::Error::AssertFail {
493                pos: heap_start,
494                message: "Mismatch between present row count and row_presence_flags".to_string(),
495            });
496        }
497
498        let rowgroup_start = writer.stream_position()? - u64::from(Self::BINARY_SIZE);
499
500        let free_space_start = heap_start + relative_row_offset;
501        const INVALID_ROW_OFFSET: u16 = u16::MAX;
502        let mut row_offsets = [INVALID_ROW_OFFSET; Self::MAX_ROW_COUNT];
503
504        // Write rows
505        writer.seek(SeekFrom::Start(free_space_start))?;
506        for (i, row) in self.present_rows().iter().enumerate() {
507            let row_position = writer.stream_position()?;
508            let aligned_position = row.align_by(row_position);
509            writer.seek(SeekFrom::Start(aligned_position))?;
510            row.write_options(writer, endian, ())?;
511
512            let large_offset = aligned_position.checked_sub(heap_start).ok_or_else(|| {
513                binrw::Error::AssertFail {
514                    pos: aligned_position,
515                    message: "Wraparound while calculating row offset".to_string(),
516                }
517            })?;
518            row_offsets[i] = large_offset
519                .try_into()
520                .map_err(|error| binrw::Error::AssertFail {
521                    pos: aligned_position,
522                    message: format!("Error converting offset: {:?}", error),
523                })?;
524        }
525        let written_space_end = writer.stream_position()?;
526        writer.seek(SeekFrom::Start(rowgroup_start))?;
527
528        // Write the offsets in reverse order, which matches the file format.
529        for offset in row_offsets.into_iter().rev() {
530            if offset == INVALID_ROW_OFFSET {
531                // Just skip the row, don't write zeros to it as
532                // there may be valid content there
533                writer.seek_relative(2)?;
534            } else {
535                offset.write_options(writer, endian, ())?;
536            }
537        }
538        self.row_presence_flags.write_options(writer, endian, ())?;
539        self.unknown.write_options(writer, endian, ())?;
540        // Seek back to the beginning of this rowgroup (which is the end of the next rowgroup)
541        writer.seek(SeekFrom::Start(rowgroup_start))?;
542
543        Ok(written_space_end - heap_start)
544    }
545}

1099    #[must_use]
1100    const fn align_by(&self, offset: u64) -> u64 {
1101        use crate::pdb::PlainRow::*;
1102        use crate::util::align_by;
1103        use std::mem::align_of_val;
1104        // unfortunately I couldn't find any less copy-pastey way of doing this
1105        // without unnecessarily complex macros.
1106        match &self {
1107            Album(_) => offset,
1108            Artist(_) => offset,
1109            Artwork(_) => align_by(4, offset),
1110            Color(_) => align_by(4, offset),
1111            ColumnEntry(r) => align_by(align_of_val(r) as u64, offset),
1112            Genre(_) => align_by(4, offset), // fixed alignment to 4 bytes
1113            HistoryPlaylist(r) => align_by(align_of_val(r) as u64, offset),

1115            Key(_) => align_by(4, offset),
1116            Label(_) => align_by(4, offset),
1117            PlaylistTreeNode(_) => align_by(4, offset),
1118            PlaylistEntry(r) => align_by(align_of_val(r) as u64, offset),
1119            Track(_) => offset, // already handled by track serialization
1120        }
1121    }
