44/// An error that can occur when parsing a PDB file.
45#[derive(Debug, Error)]
46pub enum PdbError {
47    /// An invalid value was passed when creating a `PageIndex`.
48    #[error("Invalid page index value: {0:#X}")]
49    InvalidPageIndex(u32),
50    /// Invalid flags were passed when creating an `IndexEntry`.
51    #[error("Invalid index flags (expected max 3 bits): {0:#b}")]
52    InvalidIndexFlags(u8),
53    /// A row was added to a full `RowGroup`.
54    #[error("Cannot add row to a full row group (max 16 rows)")]
55    RowGroupFull,
56}

141/// Points to a table page and can be used to calculate the page's file offset by multiplying it
142/// with the page size (found in the file header).
143#[binrw]
144#[derive(Clone, Debug, PartialEq, Eq, PartialOrd)]
145#[brw(little)]
146pub struct PageIndex(u32);
147
148impl TryFrom<u32> for PageIndex {
149    type Error = PdbError;
150
151    fn try_from(value: u32) -> Result<Self, Self::Error> {
152        if value < 0x03FF_FFFF {
153            Ok(Self(value))
154        } else {
155            Err(PdbError::InvalidPageIndex(value))
156        }
157    }
158}
159
160impl PageIndex {
161    /// Calculate the absolute file offset of the page in the PDB file for the given `page_size`.
162    #[must_use]
163    pub fn offset(&self, page_size: u32) -> u64 {
164        u64::from(self.0) * u64::from(page_size)
165    }
166}

181    empty_candidate: u32,

183    ///
184    /// *Note:* The first page apparently does not contain any rows. If the table is non-empty, the

187    /// Index of the last page that belongs to this table.
188    pub last_page: PageIndex,

191/// The PDB header structure, including the list of tables.
192#[binrw]
193#[derive(Debug, PartialEq, Eq, Clone)]
194#[brw(little)]
195#[brw(import(db_type: DatabaseType))]
196pub struct Header {
197    // Unknown purpose, perhaps an unoriginal signature, seems to always have the value 0.
198    #[brw(magic = 0u32)]
199    /// Size of a single page in bytes.
200    ///
201    /// The byte offset of a page can be calculated by multiplying a page index with this value.
202    pub page_size: u32,
203    /// Number of tables.
204    #[br(temp)]
205    #[bw(calc = tables.len().try_into().expect("too many tables"))]
206    num_tables: u32,
207    /// Unknown field, not used as any `empty_candidate`, points past end of file.
208    #[allow(dead_code)]
209    next_unused_page: PageIndex,
210    /// Unknown field.
211    #[allow(dead_code)]
212    unknown: u32,
213    /// Always incremented by at least one, sometimes by two or three.
214    pub sequence: u32,
215    // The gap seems to be always zero.
216    #[brw(magic = 0u32)]
217    /// Each table is a linked list of pages containing rows of a particular type.
218    #[br(count = num_tables, args {inner: (db_type,)})]
219    #[bw(args(db_type))]
220    pub tables: Vec<Table>,
221}
222
223impl Header {
224    /// Returns pages for the given Table.
225    pub fn read_pages<R: Read + Seek>(
226        &self,
227        reader: &mut R,
228        _: Endian,
229        args: (&PageIndex, &PageIndex, DatabaseType),
230    ) -> BinResult<Vec<Page>> {
231        let endian = Endian::Little;
232        let (first_page, last_page, db_type) = args;
233
234        let mut pages = vec![];
235        let mut page_index = first_page.clone();
236        loop {
237            let page_offset = SeekFrom::Start(page_index.offset(self.page_size));
238            reader.seek(page_offset).map_err(binrw::Error::Io)?;
239            let page = Page::read_options(reader, endian, (self.page_size, db_type))?;
240            let is_last_page = &page.page_index == last_page;
241            page_index = page.next_page.clone();
242            pages.push(page);
243
244            if is_last_page {
245                break;
246            }
247        }
248        Ok(pages)
249    }
250}

268/// An entry in an index page.
269#[binrw]
270#[derive(PartialEq, Eq, Clone, Copy)]
271#[brw(little)]
272pub struct IndexEntry(u32);
273
274impl TryFrom<(PageIndex, u8)> for IndexEntry {
275    type Error = PdbError;
276
277    fn try_from(value: (PageIndex, u8)) -> Result<Self, Self::Error> {
278        let (page_index, index_flags) = value;
279        if index_flags & 0b111 != index_flags {
280            return Err(PdbError::InvalidIndexFlags(index_flags));
281        }
282        Ok(Self((page_index.0 << 3) | (index_flags & 0b111) as u32))
283    }
284}
285
286impl IndexEntry {
287    /// Returns bits 31-3 as a `PageIndex` which points to a page containing
288    /// data rows, with `page_flags=0x34` and same `page_type` as this page.
289    pub fn page_index(&self) -> Result<PageIndex, PdbError> {
290        PageIndex::try_from(self.0 >> 3)
291    }
292
293    /// Returns the index flags from bits 2-0. Their meaning is currently
294    /// unknown.
295    #[must_use]
296    pub fn index_flags(&self) -> u8 {
297        (self.0 & 0b111) as u8
298    }
299
300    /// Returns `true` if the entry is an empty slot.
301    #[must_use]
302    pub fn is_empty(&self) -> bool {
303        self.0 == 0x1FFF_FFF8
304    }
305
306    /// Creates a new empty `IndexEntry`.
307    #[must_use]
308    pub const fn empty() -> Self {
309        Self(0x1FFF_FFF8)
310    }
311}
312
313impl fmt::Debug for IndexEntry {
314    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
315        if self.is_empty() {
316            f.debug_struct("IndexEntry")
317                .field("is_empty", &self.is_empty())
318                .finish()
319        } else {
320            f.debug_struct("IndexEntry")
321                .field("is_empty", &self.is_empty())
322                .field("page_index", &self.page_index().unwrap())
323                .field("index_flags", &self.index_flags())
324                .finish()
325        }
326    }
327}

329/// The content of an index page.
330#[binread]
331#[derive(Debug, PartialEq, Eq, Clone)]
332#[br(little)]
333pub struct IndexPageContent {
334    /// Unknown field, usually `0x1fff` or `0x0001`.
335    pub unknown_a: u16,
336    /// Unknown field, usually `0x1fff` or `0x0000`.
337    pub unknown_b: u16,
338    // Magic value `0x03ec`.
339    #[br(magic = 0x03ecu16)]
340    /// Offset where the next index entry will be written from the beginning
341    /// of the entries array, i.e. if this is 4 it means the next entry should
342    /// be written at byte `entries+4*4`. We still do not know why this value
343    /// is sometimes different than num_entries.
344    pub next_offset: u16,
345    /// Redundant page index.
346    pub page_index: PageIndex,
347    /// Redundant next page index.
348    pub next_page: PageIndex,
349    // Magic value `0x0000000003ffffff`.
350    #[br(magic = 0x0000_0000_03ff_ffffu64)]
351    /// Number of index entries in this page.
352    #[br(temp)]
353    num_entries: u16,
354    /// Points to the first empty index entry, or `0x1fff` if none.
355    ///
356    /// In real databases, this has been found to be one of three things:
357    /// 1. The same value as `num_entries`.
358    /// 2. `0x1fff`. We assume this has the same meaning as **1.**
359    /// 3. A number smaller than `num_entries`, indicating the first empty
360    /// slot.
361    pub first_empty: u16,
362    /// The index entries.
363    #[br(count = num_entries)]
364    pub entries: Vec<IndexEntry>,
365}
366
367impl BinWrite for IndexPageContent {
368    type Args<'a> = (u32,);
369
370    fn write_options<W: Write + Seek>(
371        &self,
372        writer: &mut W,
373        endian: Endian,
374        (page_size,): Self::Args<'_>,
375    ) -> BinResult<()> {
376        let page_content_start_pos = writer.stream_position()?;
377
378        self.unknown_a.write_options(writer, endian, ())?;
379        self.unknown_b.write_options(writer, endian, ())?;
380        0x03ecu16.write_options(writer, endian, ())?;
381        self.next_offset.write_options(writer, endian, ())?;
382        self.page_index.write_options(writer, endian, ())?;
383        self.next_page.write_options(writer, endian, ())?;
384        0x0000_0000_03ff_ffffu64.write_options(writer, endian, ())?;
385        (self.entries.len() as u16).write_options(writer, endian, ())?;
386        self.first_empty.write_options(writer, endian, ())?;
387
388        for entry in &self.entries {
389            entry.write_options(writer, endian, ())?;
390        }
391
392        let after_entries_pos = writer.stream_position()?;
393        let written_bytes = after_entries_pos - page_content_start_pos;
394
395        let content_size = page_size - Page::HEADER_SIZE;
396        let padding_end_offset = content_size - 20;
397
398        // Fill with empty entries (0x1ffffff8) until the last 20 bytes, which
399        // are zeroes. If https://github.com/jam1garner/binrw/issues/205 was ever
400        // fixed, this entire BinWrite implementation could possibly be removed.
401
402        if written_bytes < u64::from(padding_end_offset) {
403            let empty_entries_to_write = (u64::from(padding_end_offset) - written_bytes) / 4;
404            let empty_entry = IndexEntry::empty();
405            for _ in 0..empty_entries_to_write {
406                empty_entry.write_options(writer, endian, ())?;
407            }
408        }
409
410        let after_padding_pos = writer.stream_position()?;
411        let final_padding_bytes =
412            content_size as u64 - (after_padding_pos - page_content_start_pos);
413
414        if final_padding_bytes > 0 {
415            let zero_padding = vec![0u8; final_padding_bytes as usize];
416            writer.write_all(&zero_padding)?;
417        }
418
419        Ok(())
420    }
421}

423/// The content of a page, which can be of different types.
424#[binrw]
425#[derive(Debug, PartialEq, Clone)]
426#[br(little, import { page_flags: PageFlags, page_start_pos: u64, page_size: u32, num_rows_small: u8, page_type: PageType })]
427#[bw(little, import { page_flags: PageFlags, page_size: u32, num_rows_small: u8, page_type: PageType })]
428pub enum PageContent {
429    /// The page contains data rows.
430    #[br(pre_assert(page_flags.page_has_data()))]
431    Data(
432        #[br(args { page_start_pos, page_size, num_rows_small, page_type })]
433        #[bw(args(page_size,))]
434        DataPageContent,
435    ),
436    /// The page is an index page.
437    #[br(pre_assert(page_flags.is_index_page()))]
438    Index(#[bw(args(page_size,))] IndexPageContent),
439    /// The page is of an unknown or unsupported format.
440    Unknown,
441}
442
443impl PageContent {
444    /// Returns the data content of the page if it is a data page.
445    #[must_use]
446    pub fn into_data(self) -> Option<DataPageContent> {
447        match self {
448            PageContent::Data(data) => Some(data),
449            _ => None,
450        }
451    }
452
453    /// Returns the index content of the page if it is an index page.
454    #[must_use]
455    pub fn into_index(self) -> Option<IndexPageContent> {
456        match self {
457            PageContent::Index(index) => Some(index),
458            _ => None,
459        }
460    }
461}

463/// A table page.
464///
465/// Each page consists of a header that contains information about the type, number of rows, etc.,
466/// followed by the data section that holds the row data. Each row needs to be located using an
467/// offset found in the page footer at the end of the page.
468#[binrw]
469#[derive(Debug, PartialEq)]
470#[brw(little, import(page_size: u32, db_type: DatabaseType))]
471pub struct Page {
472    /// Stream position at the beginning of the page; used to compute heap base for standalone buffers.
473    #[br(temp, parse_with = current_offset)]
474    #[bw(ignore)]
475    page_start_pos: u64,
476    // Magic signature for pages (must be 0).
477    #[brw(magic = 0u32)]
478    /// Index of the page.
479    ///
480    /// Should match the index used for lookup and can be used to verify that the correct page was loaded.
481    pub page_index: PageIndex,
482    /// Type of information that the rows of this page contain.
483    ///
484    /// Should match the page type of the table that this page belongs to.
485    #[brw(args(db_type))]
486    pub page_type: PageType,
487    /// Index of the next page with the same page type.
488    ///
489    /// If this page is the last one of that type, the page index stored in the field will point
490    /// past the end of the file.
491    pub next_page: PageIndex,
492    /// Unknown field.
493    /// Appears to be a number between 1 and ~2500.
494    #[allow(dead_code)]
495    unknown1: u32,
496    /// Unknown field.
497    /// Appears to always be zero.
498    #[allow(dead_code)]
499    unknown2: u32,
500    /// Number of rows in this table (8-bit version).
501    ///
502    /// Used if `num_rows_large` not greater than this value and not equal to `0x1FFF`, which means
503    /// that the number of rows fits into a single byte.
504    pub num_rows_small: u8,
505    /// Unknown field.
506    /// Appears to be a multiple of 0x20; often zero.
507    ///
508    /// According to [@flesniak](https://github.com/flesniak):
509    /// > a bitmask (first track: 32)
510    #[allow(dead_code)]
511    unknown3: u8,
512    /// Unknown field.
513    /// Appears to be between 0 and ~16; often zero.
514    ///
515    /// According to [@flesniak](https://github.com/flesniak):
516    /// > often 0, sometimes larger, esp. for pages with high real_entry_count (e.g. 12 for 101 entries)
517    #[allow(dead_code)]
518    unknown4: u8,
519    /// Page flags.
520    ///
521    /// According to [@flesniak](https://github.com/flesniak):
522    /// > strange pages: 0x44, 0x64; otherwise seen: 0x24, 0x34
523    page_flags: PageFlags,
524    /// Free space in bytes in the data section of the page (excluding the row offsets in the page footer).
525    pub free_size: u16,
526    /// Used space in bytes in the data section of the page.
527    pub used_size: u16,
528    /// The content of the page.
529    #[br(args { page_flags, page_start_pos, page_size, num_rows_small, page_type })]
530    #[bw(args { page_flags: *page_flags, page_size, num_rows_small: self.num_rows_small, page_type: self.page_type })]
531    pub content: PageContent,
532}

534/// The data-containing part of a page.
535#[binread]
536#[derive(Debug, PartialEq, Clone)]
537#[br(little, import { page_start_pos: u64, page_size: u32, num_rows_small: u8, page_type: PageType })]
538pub struct DataPageContent {
539    /// Unknown field.
540    /// Often 1 or 0x1fff; also observed: 8, 27, 22, 17, 2.
541    ///
542    /// According to [@flesniak](https://github.com/flesniak):
543    /// > (0->1: 2)
544    #[allow(dead_code)]
545    unknown5: u16,
546    /// Number of rows in this table (16-bit version).
547    ///
548    /// Used when the number of rows does not fit into a single byte. In that case,`num_rows_large`
549    /// is greater than `num_rows_small`, but is not equal to `0x1FFF`.
550    pub num_rows_large: u16,
551    /// Unknown field (usually zero).
552    #[allow(dead_code)]
553    unknown6: u16,
554    /// Unknown field (usually zero).
555    ///
556    /// According to [@flesniak](https://github.com/flesniak):
557    /// > always 0, except 1 for history pages, num entries for strange pages?"
558    /// @RobinMcCorkell: I don't think this is correct, my DB only has zeros for all pages.
559    #[allow(dead_code)]
560    unknown7: u16,
561    /// Number of rows in this page.
562    ///
563    /// **Note:** This is a virtual field and not actually read from the file.
564    #[br(temp, calc = Self::calculate_num_rows(num_rows_small, num_rows_large))]
565    num_rows: u16,
566    /// Number of rows groups in this page.
567    ///
568    /// **Note:** This is a virtual field and not actually read from the file.
569    #[br(temp, calc = num_rows.div_ceil(RowGroup::MAX_ROW_COUNT as u16))]
570    num_row_groups: u16,
571    /// The offset at which the row data for this page are located.
572    ///
573    /// **Note:** This is a virtual field and not actually read from the file.
574    #[br(temp, calc = page_start_pos + u64::from(Page::HEADER_SIZE) + u64::from(Self::HEADER_SIZE))]
575    page_heap_offset: u64,
576    /// Row groups belonging to this page.
577    #[br(seek_before(SeekFrom::Current(Page::heap_padding_size(page_size, num_row_groups).into())))]
578    #[br(args {count: num_row_groups.into(), inner: (page_type, page_heap_offset)})]
579    #[br(map(|mut vec: Vec<RowGroup>| {vec.reverse(); vec}))]
580    pub row_groups: Vec<RowGroup>,
581}

583impl BinWrite for DataPageContent {
584    type Args<'a> = (u32,);
585
586    fn write_options<W: Write + Seek>(
587        &self,
588        writer: &mut W,
589        endian: Endian,
590        (page_size,): Self::Args<'_>,
591    ) -> BinResult<()> {
592        self.unknown5.write_options(writer, endian, ())?;
593        self.num_rows_large.write_options(writer, endian, ())?;
594        self.unknown6.write_options(writer, endian, ())?;
595        self.unknown7.write_options(writer, endian, ())?;
596
597        let header_end_pos = writer.stream_position()?;
598
599        let mut relative_row_offset: u64 = 0;
600
601        // Seek to the very end of the page
602        writer.seek(SeekFrom::Current(
603            (page_size - Page::HEADER_SIZE - DataPageContent::HEADER_SIZE).into(),
604        ))?;
605
606        for row_group in &self.row_groups {
607            relative_row_offset = row_group.write_options_and_get_row_offset(
608                writer,
609                endian,
610                (header_end_pos, relative_row_offset),
611            )?;
612        }
613        Ok(())
614    }

617impl DataPageContent {
618    /// Size of the page header in bytes.
619    pub const HEADER_SIZE: u32 = 0x8;
620
621    fn calculate_num_rows(num_rows_small: u8, num_rows_large: u16) -> u16 {
622        if num_rows_large > num_rows_small.into() && num_rows_large != 0x1fff {
623            num_rows_large
624        } else {
625            num_rows_small.into()
626        }
627    }
628}

630impl Page {
631    /// Size of the page header in bytes.
632    pub const HEADER_SIZE: u32 = 0x20;
633
634    /// Calculate the size of the empty space between the header and the footer.
635    fn heap_padding_size(page_size: u32, num_row_groups: u16) -> u32 {
636        // Size of all row offsets
637        let row_groups_footer_size = u32::from(num_row_groups) * RowGroup::BINARY_SIZE;
638        page_size - Self::HEADER_SIZE - DataPageContent::HEADER_SIZE - row_groups_footer_size
639    }
640}

642/// A group of row indices, which are built backwards from the end of the page. Holds up to sixteen
643/// row offsets, along with a bit mask that indicates whether each row is actually present in the
644/// table.
645#[binread]
646#[br(import(page_type: PageType, page_heap_position: u64))]
647#[derive(Debug, Clone)]
648pub struct RowGroup {
649    /// An offset which points to a row in the table, whose actual presence is controlled by one of the
650    /// bits in `row_present_flags`. This instance allows the row itself to be lazily loaded, unless it
651    /// is not present, in which case there is no content to be loaded.
652    // rustc doesn't seem to recognize that this is used below, ignore for now
653    #[allow(dead_code)]
654    row_offsets: [u16; Self::MAX_ROW_COUNT],
655    row_presence_flags: u16,
656    /// Unknown field.
657    /// Often zero, sometimes a multiple of 2, rarely something else.
658    /// When a multiple of 2, the set bit often aligns with the last present row
659    /// in the group, so maybe this is a bitset like the flags.
660    ///
661    /// E.g. for a full Artist rowgroup, this is usually zero.
662    /// For the last Artist rowgroup in the page with flags 0x003f, this is often 0x0020.
663    unknown: u16,
664    // build rows from offsets collected above
665    #[br(seek_before=SeekFrom::Start(page_heap_position))]
666    #[br(args(page_type))]
667    #[br(parse_with = binrw::file_ptr::parse_from_iter(Self::present_rows_offsets(&row_offsets, row_presence_flags)))]
668    #[br(restore_position)] // ensure the parser points to just after this instance, this is important
669    /// Access rows in this RowGroup
670    rows: Vec<Row>,
671}
672
673impl RowGroup {
674    const MAX_ROW_COUNT: usize = 16;
675    const BINARY_SIZE: u32 = (Self::MAX_ROW_COUNT as u32) * 2 + 4; // size the serialized structure
676
677    /// Get all rows present in this rowgroup
678    #[must_use]
679    pub fn present_rows(&self) -> &[Row] {
680        &self.rows
681    }
682
683    /// Add a row to this rowgroup
684    pub fn add_row(&mut self, row: Row) -> Result<(), PdbError> {
685        if self.rows.len() >= Self::MAX_ROW_COUNT {
686            return Err(PdbError::RowGroupFull);
687        }
688        self.row_presence_flags |= 1 << self.rows.len() as u16;
689        self.rows.push(row);
690        Ok(())
691    }
692
693    fn present_rows_offsets(
694        row_offsets: &[u16; Self::MAX_ROW_COUNT],
695        row_presence_flags: u16,
696    ) -> impl Iterator<Item = u16> + '_ {
697        row_offsets
698            .iter()
699            .rev()
700            .enumerate()
701            .filter_map(move |(i, row_offset)| {
702                (row_presence_flags & (1 << i) != 0).then_some(*row_offset)
703            })
704    }
705}
706
707impl PartialEq for RowGroup {
708    fn eq(&self, other: &Self) -> bool {
709        self.rows == other.rows
710    }
711}
712
713impl RowGroup {
714    // This helper function now lives in the main impl block for RowGroup
715    // Assumes we point just past the rowgroup we're trying to write.
716    fn write_options_and_get_row_offset<W: Write + Seek>(
717        &self,
718        writer: &mut W,
719        endian: Endian,
720        args: (u64, u64),
721    ) -> binrw::BinResult<u64> {
722        let (heap_start, relative_row_offset) = args;
723
724        let rows_to_write_count = self.present_rows().len();
725
726        // The number of flags set should match the number of present rows.
727        if rows_to_write_count != self.row_presence_flags.count_ones() as usize {
728            return Err(binrw::Error::AssertFail {
729                pos: heap_start,
730                message: "Mismatch between present row count and row_presence_flags".to_string(),
731            });
732        }
733
734        let rowgroup_start = writer.stream_position()? - u64::from(Self::BINARY_SIZE);
735
736        let free_space_start = heap_start + relative_row_offset;
737        const INVALID_ROW_OFFSET: u16 = u16::MAX;
738        let mut row_offsets = [INVALID_ROW_OFFSET; Self::MAX_ROW_COUNT];
739
740        // Write rows
741        writer.seek(SeekFrom::Start(free_space_start))?;
742        for (i, row) in self.present_rows().iter().enumerate() {
743            let row_position = writer.stream_position()?;
744            let aligned_position = row.align_by(row_position);
745            writer.seek(SeekFrom::Start(aligned_position))?;
746            row.write_options(writer, endian, ())?;
747
748            let large_offset = aligned_position.checked_sub(heap_start).ok_or_else(|| {
749                binrw::Error::AssertFail {
750                    pos: aligned_position,
751                    message: "Wraparound while calculating row offset".to_string(),
752                }
753            })?;
754            row_offsets[i] = large_offset
755                .try_into()
756                .map_err(|error| binrw::Error::AssertFail {
757                    pos: aligned_position,
758                    message: format!("Error converting offset: {:?}", error),
759                })?;
760        }
761        let written_space_end = writer.stream_position()?;
762        writer.seek(SeekFrom::Start(rowgroup_start))?;
763
764        // Write the offsets in reverse order, which matches the file format.
765        for offset in row_offsets.into_iter().rev() {
766            if offset == INVALID_ROW_OFFSET {
767                // Just skip the row, don't write zeros to it as
768                // there may be valid content there
769                writer.seek_relative(2)?;
770            } else {
771                offset.write_options(writer, endian, ())?;
772            }
773        }
774        self.row_presence_flags.write_options(writer, endian, ())?;
775        self.unknown.write_options(writer, endian, ())?;
776        // Seek back to the beginning of this rowgroup (which is the end of the next rowgroup)
777        writer.seek(SeekFrom::Start(rowgroup_start))?;
778
779        Ok(written_space_end - heap_start)
780    }
781}

1352        use std::mem::align_of_val;

1367            PlaylistEntry(r) => align_by(align_of_val(r) as u64, offset),

1369        }
