127/// The PDB header structure, including the list of tables.
128#[binrw]
129#[derive(Debug, PartialEq, Eq, Clone)]
130#[brw(little)]
131pub struct Header {
132    /// Unknown purpose, perhaps an unoriginal signature, seems to always have the value 0.
133    #[br(temp, assert(unknown1 == 0))]
134    #[bw(calc = 0u32)]
135    unknown1: u32,
136    /// Size of a single page in bytes.
137    ///
138    /// The byte offset of a page can be calculated by multiplying a page index with this value.
139    pub page_size: u32,
140    /// Number of tables.
141    #[br(temp)]
142    #[bw(calc = tables.len().try_into().expect("too many tables"))]
143    num_tables: u32,
144    /// Unknown field, not used as any `empty_candidate`, points past end of file.
145    #[allow(dead_code)]
146    next_unused_page: PageIndex,
147    /// Unknown field.
148    #[allow(dead_code)]
149    unknown: u32,
150    /// Always incremented by at least one, sometimes by two or three.
151    pub sequence: u32,
152    /// The gap seems to be always zero.
153    #[br(temp, assert(gap == 0))]
154    #[bw(calc = 0u32)]
155    gap: u32,
156    /// Each table is a linked list of pages containing rows of a particular type.
157    #[br(count = num_tables)]
158    pub tables: Vec<Table>,
159}
160
161impl Header {
162    /// Returns pages for the given Table.
163    pub fn read_pages<R: Read + Seek>(
164        &self,
165        reader: &mut R,
166        _: Endian,
167        args: (&PageIndex, &PageIndex),
168    ) -> BinResult<Vec<Page>> {
169        let endian = Endian::Little;
170        let (first_page, last_page) = args;
171
172        let mut pages = vec![];
173        let mut page_index = first_page.clone();
174        loop {
175            let page_offset = SeekFrom::Start(page_index.offset(self.page_size));
176            reader.seek(page_offset).map_err(binrw::Error::Io)?;
177            let page = Page::read_options(reader, endian, (self.page_size,))?;
178            let is_last_page = &page.page_index == last_page;
179            page_index = page.next_page.clone();
180            pages.push(page);
181
182            if is_last_page {
183                break;
184            }
185        }
186        Ok(pages)
187    }
188}

201fn write_page_contents<W: Write + Seek>(
202    row_groups: &Vec<RowGroup>,
203    writer: &mut W,
204    endian: Endian,
205    args: (u32,),
206) -> BinResult<()> {
207    let (page_size,) = args;
208
209    let header_end_pos = writer.stream_position()?;
210
211    let mut relative_row_offset: u64 = 0;
212
213    // Seek to the very end of the page
214    writer.seek(SeekFrom::Current((page_size - Page::HEADER_SIZE).into()))?;
215
216    for row_group in row_groups {
217        relative_row_offset = row_group.write_options_and_get_row_offset(
218            writer,
219            endian,
220            (header_end_pos, relative_row_offset),
221        )?;
222    }
223    Ok(())
224}

226/// A table page.
227///
228/// Each page consists of a header that contains information about the type, number of rows, etc.,
229/// followed by the data section that holds the row data. Each row needs to be located using an
230/// offset found in the page footer at the end of the page.
231#[binrw]
232#[derive(Debug, PartialEq)]
233#[brw(little, import(page_size: u32))]
234pub struct Page {
235    /// Stream position at the beginning of the page; used to compute heap base for standalone buffers.
236    #[br(temp, parse_with = current_offset)]
237    #[bw(ignore)]
238    page_start_pos: u64,
239    /// Magic signature for pages (must be 0).
240    #[br(temp, assert(magic == 0u32))]
241    #[bw(calc = 0u32)]
242    magic: u32,
243    /// Index of the page.
244    ///
245    /// Should match the index used for lookup and can be used to verify that the correct page was loaded.
246    pub page_index: PageIndex,
247    /// Type of information that the rows of this page contain.
248    ///
249    /// Should match the page type of the table that this page belongs to.
250    pub page_type: PageType,
251    /// Index of the next page with the same page type.
252    ///
253    /// If this page is the last one of that type, the page index stored in the field will point
254    /// past the end of the file.
255    pub next_page: PageIndex,
256    /// Unknown field.
257    #[allow(dead_code)]
258    unknown1: u32,
259    /// Unknown field.
260    #[allow(dead_code)]
261    unknown2: u32,
262    /// Number of rows in this table (8-bit version).
263    ///
264    /// Used if `num_rows_large` not greater than this value and not equal to `0x1FFF`, which means
265    /// that the number of rows fits into a single byte.
266    pub num_rows_small: u8,
267    /// Unknown field.
268    ///
269    /// According to [@flesniak](https://github.com/flesniak):
270    /// > a bitmask (first track: 32)
271    #[allow(dead_code)]
272    unknown3: u8,
273    /// Unknown field.
274    ///
275    /// According to [@flesniak](https://github.com/flesniak):
276    /// > often 0, sometimes larger, esp. for pages with high real_entry_count (e.g. 12 for 101 entries)
277    #[allow(dead_code)]
278    unknown4: u8,
279    /// Page flags.
280    ///
281    /// According to [@flesniak](https://github.com/flesniak):
282    /// > strange pages: 0x44, 0x64; otherwise seen: 0x24, 0x34
283    page_flags: PageFlags,
284    /// Free space in bytes in the data section of the page (excluding the row offsets in the page footer).
285    pub free_size: u16,
286    /// Used space in bytes in the data section of the page.
287    pub used_size: u16,
288    /// Unknown field.
289    ///
290    /// According to [@flesniak](https://github.com/flesniak):
291    /// > (0->1: 2)
292    #[allow(dead_code)]
293    unknown5: u16,
294    /// Number of rows in this table (16-bit version).
295    ///
296    /// Used when the number of rows does not fit into a single byte. In that case,`num_rows_large`
297    /// is greater than `num_rows_small`, but is not equal to `0x1FFF`.
298    pub num_rows_large: u16,
299    /// Unknown field.
300    #[allow(dead_code)]
301    unknown6: u16,
302    /// Unknown field.
303    ///
304    /// According to [@flesniak](https://github.com/flesniak):
305    /// > always 0, except 1 for history pages, num entries for strange pages?"
306    #[allow(dead_code)]
307    unknown7: u16,
308    /// Number of rows in this page.
309    ///
310    /// **Note:** This is a virtual field and not actually read from the file.
311    #[br(temp, calc = Self::calculate_num_rows(num_rows_small, num_rows_large))]
312    #[bw(ignore)]
313    num_rows: u16,
314    /// Number of rows groups in this page.
315    ///
316    /// **Note:** This is a virtual field and not actually read from the file.
317    #[br(temp, calc = num_rows.div_ceil(RowGroup::MAX_ROW_COUNT as u16))]
318    #[bw(ignore)]
319    num_row_groups: u16,
320    /// The offset at which the row data for this page are located.
321    ///
322    /// **Note:** This is a virtual field and not actually read from the file.
323    #[br(temp, calc = page_start_pos + u64::from(Self::HEADER_SIZE))]
324    #[bw(ignore)]
325    page_heap_offset: u64,
326    /// Row groups belonging to this page.
327    #[br(seek_before(SeekFrom::Current(Self::heap_padding_size(page_size, num_row_groups).into())))]
328    #[br(args {count: num_row_groups.into(), inner: (page_type, page_heap_offset)})]
329    #[br(map(|mut vec: Vec<RowGroup>| {vec.reverse(); vec}))]
330    #[br(if(num_row_groups > 0 && page_flags.page_has_data()))]
331    #[bw(write_with = write_page_contents, args(page_size))]
332    pub row_groups: Vec<RowGroup>,
333}

335impl Page {
336    /// Size of the page header in bytes.
337    pub const HEADER_SIZE: u32 = 0x28;
338
339    /// Calculate the size of the empty space between the header and the footer.
340    fn heap_padding_size(page_size: u32, num_row_groups: u16) -> u32 {
341        // Size of all row offsets
342        let row_groups_footer_size = u32::from(num_row_groups) * RowGroup::BINARY_SIZE;
343        page_size - Self::HEADER_SIZE - row_groups_footer_size
344    }
345
346    #[must_use]
347    /// Returns `true` if the page actually contains row data.
348    pub fn has_data(&self) -> bool {
349        self.page_flags.page_has_data()
350    }
351
352    #[must_use]
353    /// Number of rows on this page.
354    ///
355    /// Note that this number includes rows that have been flagged as missing by the row group.
356    pub fn num_rows(&self) -> u16 {
357        Self::calculate_num_rows(self.num_rows_small, self.num_rows_large)
358    }
359
360    fn calculate_num_rows(num_rows_small: u8, num_rows_large: u16) -> u16 {
361        if num_rows_large > num_rows_small.into() && num_rows_large != 0x1fff {
362            num_rows_large
363        } else {
364            num_rows_small.into()
365        }
366    }
367}

369/// A group of row indices, which are built backwards from the end of the page. Holds up to sixteen
370/// row offsets, along with a bit mask that indicates whether each row is actually present in the
371/// table.
372#[binread]
373#[br(import(page_type: PageType, page_heap_position: u64))]
374#[derive(Debug)]
375pub struct RowGroup {
376    /// An offset which points to a row in the table, whose actual presence is controlled by one of the
377    /// bits in `row_present_flags`. This instance allows the row itself to be lazily loaded, unless it
378    /// is not present, in which case there is no content to be loaded.
379    // rustc doesn't seem to recognize that this is used below, ignore for now
380    #[allow(dead_code)]
381    row_offsets: [u16; Self::MAX_ROW_COUNT],
382    row_presence_flags: u16,
383    /// Unknown field, probably padding.
384    ///
385    /// Apparently this is not always zero, so it might also be something different.
386    unknown: u16,
387    // build rows from offsets collected above
388    #[br(seek_before=SeekFrom::Start(page_heap_position))]
389    #[br(args(page_type))]
390    #[br(parse_with = binrw::file_ptr::parse_from_iter(Self::present_rows_offsets(&row_offsets, row_presence_flags)))]
391    #[br(restore_position)] // ensure the parser points to just after this instance, this is important
392    /// Access rows in this RowGroup
393    rows: Vec<Row>,
394}
395
396impl RowGroup {
397    const MAX_ROW_COUNT: usize = 16;
398    const BINARY_SIZE: u32 = (Self::MAX_ROW_COUNT as u32) * 2 + 4; // size the serialized structure
399
400    /// Get all rows present in this rowgroup
401    #[must_use]
402    pub fn present_rows(&self) -> &[Row] {
403        &self.rows
404    }
405    // TODO(Swiftb0y): Add a new error category for user APIs and add the correct
406    // error herer
407    #[allow(clippy::result_unit_err)]
408    /// Add a row to this rowgroup
409    pub fn add_row(&mut self, row: Row) -> Result<(), ()> {
410        if self.rows.len() >= Self::MAX_ROW_COUNT {
411            return Err(());
412        }
413        self.row_presence_flags |= 1 << self.rows.len() as u16;
414        self.rows.push(row);
415        Ok(())
416    }
417
418    fn present_rows_offsets(
419        row_offsets: &[u16; Self::MAX_ROW_COUNT],
420        row_presence_flags: u16,
421    ) -> impl Iterator<Item = u16> + '_ {
422        row_offsets
423            .iter()
424            .rev()
425            .enumerate()
426            .filter_map(move |(i, row_offset)| {
427                (row_presence_flags & (1 << i) != 0).then_some(*row_offset)
428            })
429    }
430}
431
432impl PartialEq for RowGroup {
433    fn eq(&self, other: &Self) -> bool {
434        self.rows == other.rows
435    }
436}

438impl RowGroup {
439    // This helper function now lives in the main impl block for RowGroup
440    // Assumes we point just past the rowgroup we're trying to write.
441    fn write_options_and_get_row_offset<W: Write + Seek>(
442        &self,
443        writer: &mut W,
444        endian: Endian,
445        args: (u64, u64),
446    ) -> binrw::BinResult<u64> {
447        let (heap_start, relative_row_offset) = args;
448
449        let rows_to_write_count = self.present_rows().len();
450
451        // The number of flags set should match the number of present rows.
452        if rows_to_write_count != self.row_presence_flags.count_ones() as usize {
453            return Err(binrw::Error::AssertFail {
454                pos: heap_start,
455                message: "Mismatch between present row count and row_presence_flags".to_string(),
456            });
457        }
458
459        let rowgroup_start = writer.stream_position()? - u64::from(Self::BINARY_SIZE);
460
461        let free_space_start = heap_start + relative_row_offset;
462        let mut row_offsets = [0u16; Self::MAX_ROW_COUNT];
463
464        // Write rows
465        writer.seek(SeekFrom::Start(free_space_start))?;
466        for (i, row) in self.present_rows().iter().enumerate() {
467            let row_position = writer.stream_position()?;
468            let aligned_position = row.align_by(row_position);
469            writer.seek(SeekFrom::Start(aligned_position))?;
470            row.write_options(writer, endian, ())?;
471
472            let large_offset = aligned_position.checked_sub(heap_start).ok_or_else(|| {
473                binrw::Error::AssertFail {
474                    pos: aligned_position,
475                    message: "Wraparound while calculating row offset".to_string(),
476                }
477            })?;
478            row_offsets[i] = large_offset
479                .try_into()
480                .map_err(|error| binrw::Error::AssertFail {
481                    pos: aligned_position,
482                    message: format!("Error converting offset: {:?}", error),
483                })?;
484        }
485        let written_space_end = writer.stream_position()?;
486        writer.seek(SeekFrom::Start(rowgroup_start))?;
487
488        // Write the offsets in reverse order, which matches the file format.
489        for offset in row_offsets.into_iter().rev() {
490            offset.write_options(writer, endian, ())?;
491        }
492        self.row_presence_flags.write_options(writer, endian, ())?;
493        self.unknown.write_options(writer, endian, ())?;
494        // Seek back to the beginning of this rowgroup (which is the end of the next rowgroup)
495        writer.seek(SeekFrom::Start(rowgroup_start))?;
496
497        Ok(written_space_end - heap_start)
498    }
499}

612impl Artist {
613    fn calculate_name_seek(ofs_near: u8, ofs_far: &Option<u16>) -> SeekFrom {
614        let offset: u16 = ofs_far.map_or_else(|| ofs_near.into(), |v| v - 2) - 10;

912impl BinWrite for Track {
913    type Args<'a> = ();
914
915    fn write_options<W: Write + Seek>(
916        &self,
917        writer: &mut W,
918        endian: Endian,
919        _args: Self::Args<'_>,
920    ) -> BinResult<()> {
921        debug_assert!(endian == Endian::Little);
922
923        let base_position = writer.stream_position()?;
924        self.unknown1.write_options(writer, endian, ())?;
925        self.index_shift.write_options(writer, endian, ())?;
926        self.bitmask.write_options(writer, endian, ())?;
927        self.sample_rate.write_options(writer, endian, ())?;
928        self.composer_id.write_options(writer, endian, ())?;
929        self.file_size.write_options(writer, endian, ())?;
930        self.unknown2.write_options(writer, endian, ())?;
931        self.unknown3.write_options(writer, endian, ())?;
932        self.unknown4.write_options(writer, endian, ())?;
933        self.artwork_id.write_options(writer, endian, ())?;
934        self.key_id.write_options(writer, endian, ())?;
935        self.orig_artist_id.write_options(writer, endian, ())?;
936        self.label_id.write_options(writer, endian, ())?;
937        self.remixer_id.write_options(writer, endian, ())?;
938        self.bitrate.write_options(writer, endian, ())?;
939        self.track_number.write_options(writer, endian, ())?;
940        self.tempo.write_options(writer, endian, ())?;
941        self.genre_id.write_options(writer, endian, ())?;
942        self.album_id.write_options(writer, endian, ())?;
943        self.artist_id.write_options(writer, endian, ())?;
944        self.id.write_options(writer, endian, ())?;
945        self.disc_number.write_options(writer, endian, ())?;
946        self.play_count.write_options(writer, endian, ())?;
947        self.year.write_options(writer, endian, ())?;
948        self.sample_depth.write_options(writer, endian, ())?;
949        self.duration.write_options(writer, endian, ())?;
950        self.unknown5.write_options(writer, endian, ())?;
951        self.color.write_options(writer, endian, ())?;
952        self.rating.write_options(writer, endian, ())?;
953        self.unknown6.write_options(writer, endian, ())?;
954        self.unknown7.write_options(writer, endian, ())?;
955
956        let start_of_string_section = writer.stream_position()?;
957        debug_assert_eq!(start_of_string_section - base_position, 0x5e);
958
959        // Skip offsets, because we want to write the actual strings first.
960        let mut string_offsets = [0u16; 21];
961        writer.seek(SeekFrom::Current(0x2a))?;
962        for (i, string) in [
963            &self.isrc,
964            &self.unknown_string1,
965            &self.unknown_string2,
966            &self.unknown_string3,
967            &self.unknown_string4,
968            &self.message,
969            &self.kuvo_public,
970            &self.autoload_hotcues,
971            &self.unknown_string5,
972            &self.unknown_string6,
973            &self.date_added,
974            &self.release_date,
975            &self.mix_name,
976            &self.unknown_string7,
977            &self.analyze_path,
978            &self.analyze_date,
979            &self.comment,
980            &self.title,
981            &self.unknown_string8,
982            &self.filename,
983            &self.file_path,
984        ]
985        .into_iter()
986        .enumerate()
987        {
988            let current_position = writer.stream_position()?;
989            let offset: u16 = current_position
990                .checked_sub(base_position)
991                .and_then(|v| u16::try_from(v).ok())
992                .ok_or_else(|| binrw::Error::AssertFail {
993                    pos: current_position,
994                    message: "Wraparound while calculating row offset".to_string(),
995                })?;
996            string_offsets[i] = offset;
997            string.write_options(writer, endian, ())?;
998        }
999
1000        let end_of_row = writer.stream_position()?;
1001        writer.seek(SeekFrom::Start(start_of_string_section))?;
1002        string_offsets.write_options(writer, endian, ())?;
1003        writer.seek(SeekFrom::Start(end_of_row))?;
1004
1005        // TODO(Swiftb0y): encapsulate this properly
1006        // Rows don't seem to be directly adjacent to each other
1007        // but instead have gaps in between. They probably adhere to their
1008        // member variable alignment.
1009        // I have seen gaps of 52 to 55 bytes (ending after the last char
1010        // of the previous row and the first byte of the next row).
1011        // I have 0 idea why these gaps are this big or how to accurately
1012        // guess their size.
1013        // Rows also don't have a fixed size. Their sizes seem to fluctuate
1014        // between 0 and 48 bytes in size (though the fluctuations always
1015        // were multiple of 12)
1016
1017        let mut padding_base = 0x34;
1018        // This is a heuristic that seems to match the padding behavior of the
1019        // original file for the `track_page` test case. The actual logic
1020        // is unknown.
1021        // We're assigning a different padding base for even and odd tracks
1022        if self.id.0 % 2 == 0 {
1023            padding_base += 4;
1024        }
1025        padding_base = ((end_of_row + padding_base) & !0b11) - end_of_row;
1026        writer.seek(SeekFrom::Current(padding_base as i64))?;
1027
1028        Ok(())
1029    }
1030}
