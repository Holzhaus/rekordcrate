9//! Module for reading and writing offset arrays.
10//!
11//! An offset array consists of an array of offsets, followed by data at those offsets.
12//! The offsets can be either u8 or u16, specified by the `OffsetSize` enum.
13//! The inner type `T` is constructed from an N-array of `U`, read from the positions at
14//! N offsets. The offsets are read/written in little-endian format.
15//! For example, an `OffsetArrayContainer<T, 3>` where `T` implements `OffsetArrayItems`
16//! will read/write 3 offsets, followed by 3 instances of `T::Item` at the offsets which
17//! are used to construct an inner `T`.
18//!
19//! Example:
20//! ```
21//! # use binrw::{binrw, BinRead, BinWrite};
22//! # use rekordcrate::pdb::offset_array::{OffsetArrayContainer, OffsetArrayItems};
23//! #[derive(Debug, PartialEq)]
24//! struct SingleTarget<T: BinRead + BinWrite>(T);
25//!
26//! impl<T: BinRead + BinWrite> OffsetArrayItems<1> for SingleTarget<T> {
27//!     type Item = T;
28//!
29//!     fn as_items(&self) -> [&Self::Item; 1] {
30//!         [&self.0]
31//!     }
32//!
33//!     fn from_items(items: [Self::Item; 1]) -> Self {
34//!      let [v] = items;
35//!         Self(v)
36//!     }
37//! }
38//!
39//! use rekordcrate::util::MaybeCalculated::*;
40//! let near_u8_tail = OffsetArrayContainer {
41//!     offsets: Provided([1u8].into()),
42//!     inner: SingleTarget(42u8),
43//! };
44//! ```

50/// Per-item alignment requirement when writing `OffsetArrayContainer` items with calculated offsets.
51pub trait OffsetArrayItemAlignment {
52    /// Required byte alignment of the item's start offset.
53    fn required_alignment(&self) -> u16 {
54        1
55    }
56}
57
58impl OffsetArrayItemAlignment for u8 {}
59impl OffsetArrayItemAlignment for u16 {}
60impl OffsetArrayItemAlignment for u32 {}
61impl OffsetArrayItemAlignment for () {}
62impl<const N: usize> OffsetArrayItemAlignment for [u8; N] {}

64/// Specifies whether the offsets are stored as u8 or u16.
65#[derive(Debug, Clone, Copy, PartialEq, Eq)]
66pub enum OffsetSize {
67    /// Offsets are stored as u8.
68    U8,
69    /// Offsets are stored as u16.
70    U16,
71}
72
73impl OffsetSize {
74    fn bytes(&self) -> usize {
75        match self {
76            OffsetSize::U8 => std::mem::size_of::<u8>(),
77            OffsetSize::U16 => std::mem::size_of::<u16>(),
78        }
79    }
80}

82/// A container for an array of offsets and the indirectly-addressed data at those offsets.
83/// See the module documentation for an example.
84#[derive(Debug, PartialEq, Clone, Eq)]
85pub struct OffsetArrayContainer<T, const N: usize> {
86    /// The offsets, either u8 or u16.
87    /// May be calculated from the inner data during serialization.
88    pub offsets: MaybeCalculated<OffsetArray<N>>,
89
90    /// The inner value, read/written with the offsets.
91    pub inner: T,
92}

94impl<T, const N: usize> std::ops::Deref for OffsetArrayContainer<T, N> {
95    type Target = T;
96    fn deref(&self) -> &Self::Target {
97        &(self.inner)
98    }
99}

120impl<T, const N: usize> OffsetArrayContainer<T, N> {
121    fn calculate_base(start: u64, offset: u64) -> BinResult<u64> {
122        start
123            .checked_sub(offset)
124            .ok_or_else(|| binrw::Error::AssertFail {
125                pos: start,
126                message: format!("Stream position underflow: {start}-{offset}"),
127            })
128    }
129
130    fn aligned_offset(offset: u64, align: u16) -> u64 {
131        offset.next_multiple_of(u64::from(align.max(1)))
132    }
133}

135impl<T, U, const N: usize, IA> BinRead for OffsetArrayContainer<T, N>
136where
137    T: OffsetArrayItems<N, Item = U>,
138    U: for<'a> BinRead<Args<'a> = IA>,
139    IA: Clone,
140{
141    type Args<'a> = (u64, OffsetSize, IA);
142
143    fn read_options<R: std::io::Read + std::io::Seek>(
144        reader: &mut R,
145        endian: binrw::Endian,
146        (offset, offset_size, args): Self::Args<'_>,
147    ) -> BinResult<Self> {
148        let start = reader.stream_position()?;
149        let offsets = OffsetArray::<N>::read_options(reader, endian, (offset_size,))?;
150        if !offsets.assert_offset_size_matches(offset_size) {
151            return Err(binrw::Error::AssertFail {
152                pos: start,
153                message: format!("offsetsize mismatch! {offset_size:?}"),
154            });
155        }
156        let base = Self::calculate_base(start, offset)?;
157        reader.seek(SeekFrom::Start(base))?;
158        let seeks = offsets.as_seeks();
159        let parsed = seeks.try_map_polyfill(|seek| {
160            reader.seek(seek)?;
161            let v = U::read_options(reader, endian, args.clone())?;
162            // Restore position after each item.
163            reader.seek(SeekFrom::Start(base))?;
164            Ok::<U, binrw::Error>(v)
165        })?;
166        Ok(Self {
167            offsets: MaybeCalculated::Provided(offsets),
168            inner: T::from_items(parsed),
169        })
170    }
171}

173impl<T, U, const N: usize, IA> BinWrite for OffsetArrayContainer<T, N>
174where
175    T: OffsetArrayItems<N, Item = U>,
176    U: for<'a> BinWrite<Args<'a> = IA> + OffsetArrayItemAlignment,
177    IA: Clone,
178{
179    type Args<'a> = (u64, OffsetSize, IA);
180
181    fn write_options<W: std::io::Write + std::io::Seek>(
182        &self,
183        writer: &mut W,
184        endian: binrw::Endian,
185        (offset, offset_size, args): Self::Args<'_>,
186    ) -> BinResult<()> {
187        let start = writer.stream_position()?;
188        let base = Self::calculate_base(start, offset)?;
189        let items = self.inner.as_items();
190        match &self.offsets {
191            MaybeCalculated::Calculated => {
192                let offsets_bytes = (N + 1) * offset_size.bytes();
193                writer.seek(SeekFrom::Current(offsets_bytes as i64))?;
194                let offsets_raw = items.try_map_polyfill(|v| {
195                    let current_offset = writer.stream_position()? - base;
196                    let aligned_offset =
197                        Self::aligned_offset(current_offset, v.required_alignment());
198                    let pad = aligned_offset - current_offset;
199                    if pad > 0 {
200                        for _ in 0..pad {
201                            writer.write_all(&[0u8])?;
202                        }
203                    }
204
205                    let offset = aligned_offset;
206                    v.write_options(writer, endian, args.clone())?;
207                    Ok::<_, binrw::Error>(offset)
208                })?;
209                let offsets = OffsetArray::new(offsets_raw, offset_size).map_err(|err| {
210                    binrw::Error::AssertFail {
211                        pos: start,
212                        message: format!("{err}"),
213                    }
214                })?;
215                writer.seek(SeekFrom::Start(start))?;
216                offsets.write_options(writer, endian, ())?;
217            }
218            MaybeCalculated::Provided(offsets) => {
219                if !offsets.assert_offset_size_matches(offset_size) {
220                    return Err(binrw::Error::AssertFail {
221                        pos: start,
222                        message: format!("offsetsize mismatch! {offset_size:?}"),
223                    });
224                }
225                offsets.write_options(writer, endian, ())?;
226
227                writer.seek(SeekFrom::Start(base))?;
228                let seeks = offsets.as_seeks();
229                items.zip_polyfill(seeks).try_map_polyfill(|(v, seek)| {
230                    writer.seek(seek)?;
231                    v.write_options(writer, endian, args.clone())?;
232                    // Restore position after each item.
233                    writer.seek(SeekFrom::Start(base))?;
234                    Ok::<(), binrw::Error>(())
235                })?;
236            }
237        }
238        Ok(())
239    }
240}

242impl<T, U, const N: usize> PageHeapObject for OffsetArrayContainer<T, N>
243where
244    T: OffsetArrayItems<N, Item = U> + 'static,
245    U: for<'a> PageHeapObject<Args<'a> = ()> + OffsetArrayItemAlignment,
246{
247    type Args<'a> = OffsetSize;
248    fn heap_bytes_required(&self, offset_size: OffsetSize) -> u16 {
249        let mut total = ((N + 1) * offset_size.bytes()) as u16;
250        for item in self.inner.as_items() {
251            if matches!(self.offsets, MaybeCalculated::Calculated) {
252                total = Self::aligned_offset(u64::from(total), item.required_alignment()) as u16;
253            }
254            total += item.heap_bytes_required(());
255        }
256        total
257    }
258}

260/// Inner type that can be stored in an OffsetArrayContainer.
261pub trait OffsetArrayItems<const N: usize> {
262    /// Type of the items pointed to by the offsets.
263    type Item;
264
265    /// Returns the items as an array of references.
266    fn as_items(&self) -> [&Self::Item; N];
267
268    /// Constructs the composite type from an array of items.
269    fn from_items(items: [Self::Item; N]) -> Self;
270}
271
272impl OffsetArrayItems<0> for () {
273    type Item = ();
274
275    fn as_items(&self) -> [&Self::Item; 0] {
276        []
277    }
278
279    fn from_items(_items: [Self::Item; 0]) -> Self {}
280}

282/// The implementation of the offset array, which can be either u8 or u16.
283/// This is a private implementation detail, use `OffsetArrayContainer` instead.
284/// This enum is used to read/write the offsets, and to provide the `read_offset` and
285/// `write_offset` methods to read/write the inner type `T` at the specified offsets.
286/// The offsets are stored in little-endian format.
287///
288/// Offsets are always preceded by a magic number 0x03u8/0x0003u16.
289#[binrw]
290#[derive(Debug, Clone, PartialEq, Eq)]
291#[brw(little)]
292#[br(import(size: OffsetSize))]
293pub enum OffsetArray<const N: usize> {
294    /// Offsets are stored as u8.
295    #[br(pre_assert(size == OffsetSize::U8))]
296    U8(#[brw(magic(0x03u8))] [u8; N]),
297    /// Offsets are stored as u16.
298    #[br(pre_assert(size == OffsetSize::U16))]
299    U16(#[brw(magic(0x0003u16))] [u16; N]),
300}
301
302impl<const N: usize> PageHeapObject for OffsetArray<N> {
303    type Args<'a> = OffsetSize;
304    fn heap_bytes_required(&self, offset_size: OffsetSize) -> u16 {
305        ((N + 1) * offset_size.bytes()) as u16
306    }
307}
308
309impl<const N: usize> OffsetArray<N> {
310    fn assert_offset_size_matches(&self, offset_size: OffsetSize) -> bool {
311        matches!(
312            (self, offset_size),
313            (Self::U8(_), OffsetSize::U8) | (Self::U16(_), OffsetSize::U16)
314        )
315    }
316
317    fn as_seeks(&self) -> [SeekFrom; N] {
318        match self {
319            OffsetArray::U8(offsets) => offsets.map(|offset| SeekFrom::Current(offset.into())),
320            OffsetArray::U16(offsets) => offsets.map(|offset| SeekFrom::Current(offset.into())),
321        }
322    }
323
324    fn new(offsets: [u64; N], offset_size: OffsetSize) -> Result<Self, std::num::TryFromIntError> {
325        Ok(match offset_size {
326            OffsetSize::U8 => OffsetArray::U8(offsets.try_map_polyfill(|o| o.try_into())?),
327            OffsetSize::U16 => OffsetArray::U16(offsets.try_map_polyfill(|o| o.try_into())?),
328        })
329    }
330}
331
332impl<const N: usize> From<[u8; N]> for OffsetArray<N> {
333    fn from(arr: [u8; N]) -> Self {
334        Self::U8(arr)
335    }
336}
337
338impl<const N: usize> From<[u16; N]> for OffsetArray<N> {
339    fn from(arr: [u16; N]) -> Self {
340        Self::U16(arr)
341    }
342}
