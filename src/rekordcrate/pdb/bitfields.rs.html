9//! Packed fields/bitfields used in PDB files.
10
11// This is necessary since `bitfield` generates methods that otherwise violate this lint.
12#![allow(clippy::must_use_candidate)]
13
14use binrw::{BinRead, BinWrite};
15use modular_bitfield::prelude::*;
16
17use crate::pdb::RowGroup;
18
19/// Packed field found in the page header containing:
20/// - number of used row offsets in the page (13 bits).
21/// - number of valid rows in the page (11 bits).
22#[bitfield]
23#[derive(BinRead, BinWrite, Debug, PartialEq, Eq, Clone, Copy)]
24#[br(little, map = Self::from_bytes)]
25#[bw(little, map = |x: &PackedRowCounts| x.into_bytes())]
26pub struct PackedRowCounts {
27    pub num_rows: B13,
28    pub num_rows_valid: B11,
29}
30
31impl Default for PackedRowCounts {
32    fn default() -> Self {
33        Self::new()
34    }
35}
36
37impl PackedRowCounts {
38    /// Create a `PackedRowCounts` assuming all rows in the page are valid,
39    /// e.g. when we are serializing a page without any deleted rows.
40    pub fn from_all_valid(num_rows: usize) -> Self {
41        Self::new()
42            .with_num_rows(num_rows as u16)
43            .with_num_rows_valid(num_rows as u16)
44    }
45
46    /// Get the number of row groups in the page.
47    pub(crate) fn num_row_groups(&self) -> u16 {
48        self.num_rows().div_ceil(RowGroup::MAX_ROW_COUNT as u16)
49    }
50}
