9//! Parser for the extended Pioneer DeviceSQL database exports (exportExt.pdb).
10//!
11//! The Rekordbox DJ software uses writes PDB files to `/PIONEER/rekordbox/exportExt.pdb`.
12//!
13//! Most of the base file format has been reverse-engineered by Henry Betts, Fabian Lesniak and James
14//! Elliott.
15//! The exportExt specifics have been reversed engineered by Dominik Stolz (@voidc).
16//!
17//! - <https://github.com/Deep-Symmetry/crate-digger/blob/master/doc/Analysis.pdf>
18//! - <https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html>
19//! - <https://github.com/henrybetts/Rekordbox-Decoding>
20//! - <https://github.com/flesniak/python-prodj-link/tree/master/prodj/pdblib>

29/// A unique identifier for a tag or category.
30#[binrw]
31#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
32#[brw(little)]
33pub struct TagId(pub u32);
34
35impl PageHeapObject for TagId {
36    type Args<'a> = ();
37    fn heap_bytes_required(&self, _: ()) -> u16 {
38        self.0.heap_bytes_required(())
39    }
40}

42#[binrw]
43#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
44#[brw(little)]
45/// A possibly absent parent ID. If the ID is zero (None), then there is no parent.
46pub struct ParentId(
47    #[br(try)] // failing to parse is fine, since then its just non-zero
48    #[bw(map = |&x| x.map_or(0, |v| v.get()))]
49    pub Option<NonZero<u32>>,
50);
51
52impl PageHeapObject for ParentId {
53    type Args<'a> = ();
54    fn heap_bytes_required(&self, _: ()) -> u16 {
55        let inner: u32 = self.0.map_or(0u32, |v| v.get());
56        inner.heap_bytes_required(())
57    }
58}

60#[derive(Debug, PartialEq, Clone, Eq)]
61/// The strings associated with a tag or category.
62pub struct TagOrCategoryStrings {
63    /// The name of the tag or category.
64    pub name: DeviceSQLString,
65    /// String with unknown purpose, often empty.
66    pub unknown: DeviceSQLString,
67}
68
69impl OffsetArrayItems<2> for TagOrCategoryStrings {
70    type Item = DeviceSQLString;
71
72    fn as_items(&self) -> [&Self::Item; 2] {
73        [&self.name, &self.unknown]
74    }
75
76    fn from_items(items: [Self::Item; 2]) -> Self {
77        let [name, unknown] = items;
78        Self { name, unknown }
79    }
80}

82/// A tag or category that can be assigned to tracks for the purpose of categorization.
83///
84/// ## References
85///
86/// - <https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html#tag-rows>
87#[binrw]
88#[derive(Debug, PartialEq, Eq, Clone)]
89#[brw(little)]
90pub struct TagOrCategory {
91    /// Determines if an 8-bit offset (0x0680) or a 16-bit offset (0x0684) is used for the strings.
92    pub subtype: Subtype,
93    // also called tag_index. Seems to increment by 0x20 every row.
94    /// Incrementing index (0x20 for each row).
95    pub index_shift: u16,
96    // no idea what these two do, but they aren't always zero
97    // as described on https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html#tag-rows
98    /// Unknown purpose.
99    pub unknown1: u32,
100    /// Unknown purpose.
101    pub unknown2: u32,
102    /// The ID of the parent category, if any.
103    pub parent_id: ParentId,
104    /// Zero-based position at which this tag should be displayed within its category.
105    /// If the row represents a category rather than a tag, then this is the zero-based
106    /// position of the category itself within the category list.
107    pub position: u32,
108    /// Numeric ID of the tag or category.
109    pub id: TagId,
110    /// Non-zero if this row represents a category rather than a tag.
111    pub raw_is_category: u32,
112    /// The strings associated with this tag or category.
113    #[brw(args(0x1C, subtype.get_offset_size(), ()))]
114    pub offsets: OffsetArrayContainer<TagOrCategoryStrings, 2>,
115}
116
117impl PageHeapObject for TagOrCategory {
118    type Args<'a> = ();
119    fn heap_bytes_required(&self, _: ()) -> u16 {
120        [
121            self.subtype.heap_bytes_required(()),
122            self.index_shift.heap_bytes_required(()),
123            self.unknown1.heap_bytes_required(()),
124            self.unknown2.heap_bytes_required(()),
125            self.parent_id.heap_bytes_required(()),
126            self.position.heap_bytes_required(()),
127            self.id.heap_bytes_required(()),
128            self.raw_is_category.heap_bytes_required(()),
129            self.offsets
130                .heap_bytes_required(self.subtype.get_offset_size()),
131        ]
132        .iter()
133        .sum()
134    }
135}

137// https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html#tag-track-rows
138/// M*N junction table between tags and tracks.
139#[binrw]
140#[derive(Debug, PartialEq, Eq, Clone)]
141#[brw(little)]
142pub struct TrackTag {
143    #[brw(magic(0u32))]
144    /// The ID of the track.
145    pub track_id: TrackId,
146    /// The ID of the tag.
147    pub tag_id: TagId,
148    /// Unknown purpose, seems to be always 3.
149    pub unknown_const: u32, // always 3?
150}
151
152impl PageHeapObject for TrackTag {
153    type Args<'a> = ();
154    fn heap_bytes_required(&self, _: ()) -> u16 {
155        [
156            (0u32).heap_bytes_required(()),
157            self.track_id.heap_bytes_required(()),
158            self.tag_id.heap_bytes_required(()),
159            self.unknown_const.heap_bytes_required(()),
160        ]
161        .iter()
162        .sum()
163    }
164}

166/// The type of ext pages found inside a `Table`.
167#[binrw]
168#[brw(little)]
169#[derive(Debug, PartialEq, Eq, Clone, Copy)]
170pub enum ExtPageType {
171    /// can be assigned to tracks for the purpose of categorization.
172    #[brw(magic = 3u32)]
173    Tag,
174    /// holds the associations between tag ids and track ids.
175    #[brw(magic = 4u32)]
176    TrackTag,
177}
178
179/// A table row contains the actual data.
180#[binrw]
181#[derive(Debug, PartialEq, Eq, Clone)]
182#[brw(little)]
183// #[br(import(page_type: PageType))]
184// The large enum size is unfortunate, but since users of this library will probably use iterators
185// to consume the results on demand, we can live with this. The alternative of using a `Box` would
186// require a heap allocation per row, which is arguably worse. Hence, the warning is disabled for
187// this enum.
188#[br(import(page_type: ExtPageType))]
189pub enum ExtRow {
190    /// Contains the album name, along with an ID of the corresponding artist.
191    #[br(pre_assert(page_type == ExtPageType::Tag))]
192    Tag(#[bw(align_after = 4)] TagOrCategory),
193    /// Contains the artist name and ID.
194    #[br(pre_assert(page_type == ExtPageType::TrackTag))]
195    TrackTag(#[bw(align_after = 4)] TrackTag),
196}
197
198impl PageHeapObject for ExtRow {
199    type Args<'a> = ();
200    fn heap_bytes_required(&self, _: ()) -> u16 {
201        match self {
202            ExtRow::Tag(tag_or_category) => tag_or_category.heap_bytes_required(()),
203            ExtRow::TrackTag(track_tag) => track_tag.heap_bytes_required(()),
204        }
205    }
206}
