9//! Parser for the extended Pioneer DeviceSQL database exports (exportExt.pdb).
10//!
11//! The Rekordbox DJ software uses writes PDB files to `/PIONEER/rekordbox/exportExt.pdb`.
12//!
13//! Most of the base file format has been reverse-engineered by Henry Betts, Fabian Lesniak and James
14//! Elliott.
15//! The exportExt specifics have been reversed engineered by Dominik Stolz (@voidc).
16//!
17//! - <https://github.com/Deep-Symmetry/crate-digger/blob/master/doc/Analysis.pdf>
18//! - <https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html>
19//! - <https://github.com/henrybetts/Rekordbox-Decoding>
20//! - <https://github.com/flesniak/python-prodj-link/tree/master/prodj/pdblib>
21
22use crate::pdb::{DeviceSQLString, OffsetArray, OffsetArrayContainer, Subtype, TrackId};
23use binrw::binrw;
24use std::num::NonZero;

26/// A unique identifier for a tag or category.
27#[binrw]
28#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
29#[brw(little)]
30pub struct TagId(pub u32);
31
32#[binrw]
33#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
34#[brw(little)]
35/// A possibly absent parent ID. If the ID is zero (None), then there is no parent.
36pub struct ParentId(
37    #[br(try)] // failing to parse is fine, since then its just non-zero
38    #[bw(map = |&x| x.map_or(0, |v| v.get()))]
39    pub Option<NonZero<u32>>,
40);
41
42#[binrw]
43#[brw(little)]
44#[brw(import(base: i64, offsets: &OffsetArray<3>, args: ()))]
45#[derive(Debug, PartialEq, Clone, Eq)]
46/// The strings associated with a tag or category.
47pub struct TagOrCategoryStrings {
48    #[brw(args(base, args))]
49    #[br(parse_with = offsets.read_offset(1))]
50    #[bw(write_with = offsets.write_offset(1))]
51    /// The name of the tag or category.
52    pub name: DeviceSQLString,
53    #[brw(args(base, args))]
54    #[br(parse_with = offsets.read_offset(2))]
55    #[bw(write_with = offsets.write_offset(2))]
56    /// String with unknown purpose, often empty.
57    pub unknown: DeviceSQLString,
58}

60/// A tag or category that can be assigned to tracks for the purpose of categorization.
61///
62/// ## References
63///
64/// - <https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html#tag-rows>
65#[binrw]
66#[derive(Debug, PartialEq, Eq, Clone)]
67#[brw(little)]
68pub struct TagOrCategory {
69    /// Determines if an 8-bit offset (0x0680) or a 16-bit offset (0x0684) is used for the strings.
70    pub subtype: Subtype,
71    // also called tag_index. Seems to increment by 0x20 every row.
72    /// Incrementing index (0x20 for each row).
73    pub index_shift: u16,
74    // no idea what these two do, but they aren't always zero
75    // as described on https://djl-analysis.deepsymmetry.org/rekordbox-export-analysis/exports.html#tag-rows
76    /// Unknown purpose.
77    pub unknown1: u32,
78    /// Unknown purpose.
79    pub unknown2: u32,
80    /// The ID of the parent category, if any.
81    pub parent_id: ParentId,
82    /// Zero-based position at which this tag should be displayed within its category.
83    /// If the row represents a category rather than a tag, then this is the zero-based
84    /// position of the category itself within the category list.
85    pub position: u32,
86    /// Numeric ID of the tag or category.
87    pub id: TagId,
88    /// Non-zero if this row represents a category rather than a tag.
89    pub raw_is_category: u32,
90    // Padded at the end by 11 bytes as observed
91    #[brw(args(0x1C, subtype.get_offset_size(), ()), pad_after = 11)]
92    /// The strings associated with this tag or category.
93    pub offsets: OffsetArrayContainer<TagOrCategoryStrings, 3>,
94}

97/// M*N junction table between tags and tracks.
98#[binrw]
99#[derive(Debug, PartialEq, Eq, Clone)]
100#[brw(little)]
101pub struct TrackTag {
102    #[brw(magic(0u32))]
103    /// The ID of the track.
104    pub track_id: TrackId,
105    /// The ID of the tag.
106    pub tag_id: TagId,
107    /// Unknown purpose, seems to be always 3.
108    pub unknown_const: u32, // always 3?
109}

111/// The type of ext pages found inside a `Table`.
112#[binrw]
113#[brw(little)]
114#[derive(Debug, PartialEq, Eq, Clone, Copy)]
115pub enum ExtPageType {
116    /// can be assigned to tracks for the purpose of categorization.
117    #[brw(magic = 3u32)]
118    Tag,
119    /// holds the associations between tag ids and track ids.
120    #[brw(magic = 4u32)]
121    TrackTag,
122}
123
124/// A table row contains the actual data.
125#[binrw]
126#[derive(Debug, PartialEq, Eq, Clone)]
127#[brw(little)]
128// #[br(import(page_type: PageType))]
129// The large enum size is unfortunate, but since users of this library will probably use iterators
130// to consume the results on demand, we can live with this. The alternative of using a `Box` would
131// require a heap allocation per row, which is arguably worse. Hence, the warning is disabled for
132// this enum.
133#[br(import(page_type: ExtPageType))]
134pub enum ExtRow {
135    /// Contains the album name, along with an ID of the corresponding artist.
136    #[br(pre_assert(page_type == ExtPageType::Tag))]
137    Tag(TagOrCategory),
138    /// Contains the artist name and ID.
139    #[br(pre_assert(page_type == ExtPageType::TrackTag))]
140    TrackTag(TrackTag),
141}
142
143impl ExtRow {
144    #[must_use]
145    pub(crate) const fn align_by(&self, offset: u64) -> u64 {
146        use crate::util::align_by;
147        use std::mem::align_of_val;
148        match self {
149            ExtRow::Tag(_) => align_by(4, offset),
150            ExtRow::TrackTag(r) => align_by(align_of_val(r) as u64, offset),
151        }
152    }
153}
