9//! Wrapper types for dealing with XOR encryption.
10
11use binrw::io::{Read, Result, Seek, SeekFrom, Write};
12use std::iter::Cycle;
13
14/// Stream cipher wrapper around another [`Read`]- or [`Write`]-able stream that XOR's all bytes
15/// after reading/before writing.
16#[derive(Debug)]
17pub struct XorStream<T> {
18    stream: T,
19    key: Cycle<std::vec::IntoIter<u8>>,
20    key_size: u64,
21}
22
23impl<T> XorStream<T> {
24    /// Create a new XOR wrapper around `stream` that passes XOR's all data with `key` before
25    /// forwarding it.
26    pub fn with_key(stream: T, key: Vec<u8>) -> Self {
27        let key = if key.is_empty() { vec![0] } else { key };
28        let key_size = key.len() as u64;
29        let key = key.into_iter().cycle();
30        Self {
31            stream,
32            key,
33            key_size,
34        }
35    }
36}
37
38impl<R: Read> Read for XorStream<R> {
39    fn read(&mut self, buf: &mut [u8]) -> Result<usize> {
40        let bytes_read = self.stream.read(buf)?;
41
42        for byte in buf {
43            let key_byte = self.key.next().unwrap_or(0);
44            *byte ^= key_byte;
45        }
46
47        Ok(bytes_read)
48    }
49}
50
51impl<W: Write> Write for XorStream<W> {
52    fn write(&mut self, buf: &[u8]) -> Result<usize> {
53        let encrypted_buffer: Vec<u8> = buf
54            .iter()
55            .map(|x| {
56                let key_byte = self.key.next().unwrap_or(0);
57                x ^ key_byte
58            })
59            .collect();
60
61        self.stream.write(&encrypted_buffer)
62    }
63
64    fn flush(&mut self) -> Result<()> {
65        self.stream.flush()
66    }
67}
68
69impl<S: Seek> Seek for XorStream<S> {
70    fn seek(&mut self, position: SeekFrom) -> Result<u64> {
71        let old_position = self.stream.stream_position()?;
72        let new_position = self.stream.seek(position)?;
73
74        // Calculate how many bytes we need to move forward in the key stream to match the seek in
75        // the actual buffer.
76        let offset = if new_position > old_position {
77            (new_position - old_position) % self.key_size
78        } else {
79            self.key_size - ((old_position - new_position) % self.key_size)
80        };
81
82        for _ in 0..offset {
83            self.key.next();
84        }
85
86        Ok(new_position)
87    }
88}
